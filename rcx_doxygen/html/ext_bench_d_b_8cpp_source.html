69 uint extMain::genExtRules(const char* rulesFileName, int pattern)
70 {
71  uint widthCnt = 12;
72  uint layerCnt = _tech->getRoutingLayerCount() + 1;
73 
74  extRCModel* extRulesModel = new extRCModel(layerCnt, "TYPICAL", logger_);
75  this->_modelTable->add(extRulesModel);
76  extRulesModel->setDiagModel(1);
77  extRulesModel->setOptions("./", "", false, true, false, false);
78 
79  extMetRCTable* rcModel = extRulesModel->initCapTables(layerCnt, widthCnt);
80 
81  AthPool<extDistRC>* rcPool = rcModel->getRCPool();
82  extMeasure m(logger_);
83 
84  m._diagModel = 1;
85 
86  char buff[2000];
87  sprintf(buff, "%s.log", rulesFileName);
88  FILE* logFP = fopen(buff, "w");
89 
90  Ath__parser* p = new Ath__parser(logger_);
91  Ath__parser* w = new Ath__parser(logger_);
92 
93  int prev_sep = 0;
94  int prev_width = 0;
95  int n = 0;
96 
97  dbSet<dbNet> nets = _block->getNets();
98  dbSet<dbNet>::iterator itr;
99  for (itr = nets.begin(); itr != nets.end(); ++itr) {
100  dbNet* net = *itr;
101  const char* netName = net->getConstName();
102 
103  uint wireCnt = 0;
104  uint viaCnt = 0;
105  uint len = 0;
106  uint layerCnt = 0;
107  uint layerTable[20];
108 
109  net->getNetStats(wireCnt, viaCnt, len, layerCnt, layerTable);
110 
111  uint wcnt = p->mkWords(netName, "_");
112  if (wcnt < 5) {
113  continue;
114  }
115 
116  int targetWire = 0;
117  if (p->getFirstChar() == 'U') {
118  targetWire = p->getInt(0, 1);
119  } else {
120  char* w1 = p->get(0);
121  if (w1[1] == 'U') { // OU
122  targetWire = p->getInt(0, 2);
123  } else {
124  targetWire = p->getInt(0, 1);
125  }
126  }
127  if (targetWire <= 0) {
128  continue;
129  }
130 
131  uint wireNum = p->getInt(4);
132  if (wireNum != targetWire / 2) {
133  continue;
134  }
135 
136  bool diag = false;
137  bool ResModel = false;
138 
139  uint met = p->getInt(0, 1);
140  uint overMet = 0;
141  uint underMet = 0;
142 
143  m._overMet = -1;
144  m._underMet = -1;
145  m._overUnder = false;
146  m._over = false;
147  m._diag = false;
148  m._res = false;
149 
150  char* overUnderToken = p->get(1); // M2oM1uM3
151  int wCnt = w->mkWords(overUnderToken, "ou");
152  if (wCnt < 2) {
153  continue;
154  }
155 
156  if (wCnt == 3) { // M2oM1uM3
157  met = w->getInt(0, 1);
158  overMet = w->getInt(1, 1);
159  underMet = w->getInt(2, 1);
160 
161  m._overMet = underMet;
162  m._underMet = overMet;
163  m._overUnder = true;
164  m._over = false;
165 
166  } else if (strstr(overUnderToken, "o") != nullptr) {
167  met = w->getInt(0, 1);
168  overMet = w->getInt(1, 1);
169  if (p->getFirstChar() == 'R') {
170  ResModel = true;
171  m._res = ResModel;
172  }
173  m._overMet = -1;
174  m._underMet = overMet;
175  m._overUnder = false;
176  m._over = true;
177  } else if (strstr(overUnderToken, "uu") != nullptr) {
178  met = w->getInt(0, 1);
179  underMet = w->getInt(1, 1);
180  diag = true;
181  m._diag = true;
182  m._overMet = underMet;
183  m._underMet = -1;
184  m._overUnder = false;
185  m._over = false;
186  } else if (strstr(overUnderToken, "u") != nullptr) {
187  met = w->getInt(0, 1);
188  underMet = w->getInt(1, 1);
189 
190  m._overMet = underMet;
191  m._underMet = -1;
192  m._overUnder = false;
193  m._over = false;
194  }
195  m._met = met;
196 
197  if (w->mkWords(p->get(2), "W") <= 0) {
198  continue;
199  }
200 
201  double w1 = w->getDouble(0) / 1000;
202 
203  m._w_m = w1;
204  m._w_nm = lround(m._w_m * 1000);
205 
206  if (w->mkWords(p->get(3), "S") <= 0) {
207  continue;
208  }
209 
210  double s1 = w->getDouble(0) / 1000;
211  double s2 = w->getDouble(1) / 1000;
212 
213  m._s_m = s1;
214  m._s_nm = lround(m._s_m * 1000);
215  m._s2_m = s2;
216  m._s2_nm = lround(m._s2_m * 1000);
217 
218  double wLen = GetDBcoords2(len) * 1.0;
219  double totCC = net->getTotalCouplingCap();
220  double totGnd = net->getTotalCapacitance();
221  double res = net->getTotalResistance();
222 
223  double contextCoupling = getTotalCouplingCap(net, "cntxM", 0);
224  if (contextCoupling > 0) {
225  totGnd += contextCoupling;
226  totCC -= contextCoupling;
227  }
228 
229  double cc = totCC / wLen / 2;
230  double gnd = totGnd / wLen / 2;
231  double R = res / wLen / 2;
232 
233  extDistRC* rc = rcPool->alloc();
234 
235  if (ResModel) {
236  R *= 2;
237  rc->set(m._s_nm, m._s2_m, 0.0, 0.0, R);
238  } else {
239  if (diag) {
240  rc->set(m._s_nm, 0.0, cc, cc, R);
241  } else {
242  if (m._s_nm == 0) {
243  m._s_nm = prev_sep + prev_width;
244  }
245  rc->set(m._s_nm, cc, gnd, 0.0, R);
246  }
247  }
248  m._tmpRC = rc;
249  rcModel->addRCw(&m);
250  prev_sep = m._s_nm;
251  prev_width = m._w_nm;
252 
253  if (ResModel) {
254  fprintf(
255  logFP,
256  "M%2d OVER %2d UNDER %2d W %.3f S1 %.3f S2 %.3f R %g LEN %g %g %s\n",
257  met,
258  overMet,
259  underMet,
260  w1,
261  s1,
262  s2,
263  res,
264  wLen,
265  R,
266  netName);
267  } else {
268  fprintf(
269  logFP,
270  "M%2d OVER %2d UNDER %2d W %.3f S %.3f CC %.6f GND %.6f TC %.6f x "
271  "%.6f R %g LEN %g %s\n",
272  met,
273  overMet,
274  underMet,
275  w1,
276  s1,
277  totCC,
278  totGnd,
279  totCC + totGnd,
280  contextCoupling,
281  res,
282  wLen,
283  netName);
284  }
285  }
286  rcModel->mkWidthAndSpaceMappings();
287  extRulesModel->writeRules((char*) rulesFileName, false);
288 
289  fclose(logFP);
290  return n;
291 }

293 double extMain::getTotalCouplingCap(dbNet* net,
294  const char* filterNet,
295  uint corner)
296 {
297  double cap = 0.0;
298  for (dbCapNode* n : net->getCapNodes()) {
299  for (dbCCSeg* cc : n->getCCSegs()) {
300  dbNet* srcNet = cc->getSourceCapNode()->getNet();
301  dbNet* tgtNet = cc->getTargetCapNode()->getNet();
302  if ((strstr(srcNet->getConstName(), filterNet) == nullptr)
303  && (strstr(tgtNet->getConstName(), filterNet) == nullptr)) {
304  continue;
305  }
306 
307  cap += cc->getCapacitance(corner);
308  }
309  }
310  return cap;
311 }

387 uint extRCModel::benchDB_WS(extMainOptions* opt, extMeasure* measure)
388 {
389  Ath__array1D<double>* widthTable = new Ath__array1D<double>(4);
390  Ath__array1D<double>* spaceTable = new Ath__array1D<double>(4);
391  Ath__array1D<double>* wTable = &opt->_widthTable;
392  Ath__array1D<double>* sTable = &opt->_spaceTable;
393  Ath__array1D<double>* gTable = &opt->_gridTable;
394 
395  uint cnt = 0;
396  int met = measure->_met;
397  dbTechLayer* layer = opt->_tech->findRoutingLayer(met);
398 
399  double minWidth = 0.001 * layer->getWidth();
400  double spacing = 0.001 * layer->getSpacing();
401  double pitch = 0.001 * layer->getPitch();
402  if (layer->getSpacing() == 0) {
403  spacing = pitch - minWidth;
404  }
405 
406  if (opt->_default_lef_rules) { // minWidth, minSpacing, minThickness, pitch
407  // multiplied by grid_list
408  for (uint ii = 0; ii < gTable->getCnt(); ii++) {
409  double s = spacing * gTable->get(ii);
410  spaceTable->add(s);
411  double w = minWidth * gTable->get(ii);
412  widthTable->add(w);
413  }
414  } else if (opt->_nondefault_lef_rules) {
415  return 0;
416  } else {
417  if (measure->_diag) {
418  spaceTable->add(0.0);
419  }
420  if (!opt->_res_patterns) {
421  for (uint ii = 0; ii < sTable->getCnt(); ii++) {
422  double s = spacing * sTable->get(ii);
423  if (sTable->get(ii) == 0 && measure->_diag) {
424  continue;
425  }
426  spaceTable->add(s);
427  }
428  } else {
429  spaceTable->add(0);
430 
431  for (uint ii = 1; ii < 2; ii++) {
432  double s = pitch * ii;
433  double s1 = s - minWidth;
434 
435  spaceTable->add(s1);
436  spaceTable->add(s);
437  }
438  for (uint ii = 2; ii < 4; ii++) {
439  double s = pitch * ii;
440  double s1 = s - minWidth;
441  double s2 = s1 - minWidth / 2;
442 
443  spaceTable->add(s2);
444  spaceTable->add(s1);
445  spaceTable->add(s);
446  }
447  spaceTable->add(pitch * 4);
448  }
449  for (uint ii = 0; ii < wTable->getCnt(); ii++) {
450  double w = minWidth * wTable->get(ii);
451  widthTable->add(w);
452  }
453  }
454  bool use_symmetric_widths_spacings = false;
455  if (!use_symmetric_widths_spacings) {
456  for (uint ii = 0; ii < widthTable->getCnt(); ii++) {
457  double w = widthTable->get(ii); // layout
458  double w2 = w;
459  if (!opt->_res_patterns) {
460  for (uint jj = 0; jj < spaceTable->getCnt(); jj++) {
461  double s = spaceTable->get(jj); // layout
462  double s2 = s;
463 
464  measure->setTargetParams(w, s, 0.0, 0, 0, w2, s2);
465  writeBenchWires_DB(measure);
466 
467  cnt++;
468  }
469  } else {
470  for (uint jj = 0; jj < spaceTable->getCnt(); jj++) {
471  double s = spaceTable->get(jj); // layout
472  for (uint kk = jj; kk < spaceTable->getCnt(); kk++) {
473  double s2 = spaceTable->get(kk);
474 
475  measure->setTargetParams(w, s, 0.0, 0, 0, w2, s2);
476  writeBenchWires_DB_res(measure);
477 
478  cnt++;
479  }
480  }
481  }
482  }
483  }
484  return cnt;
485 }

577 int extRCModel::writeBenchWires_DB(extMeasure* measure)

581  }

588  int bboxLL[2];
589  bboxLL[measure->_dir] = measure->_ur[measure->_dir];
590  bboxLL[!measure->_dir] = measure->_ll[!measure->_dir];
591 
592  // ASSUME odd number of wires, 2 will also work
593  int n = measure->_wireCnt / 2;
594 
595  if (measure->_s_nm == 0 && !measure->_diag) {
596  n = 1;
597  }
598 
599  uint w_layout = measure->_minWidth;
600  uint s_layout = measure->_minSpace;
601 
602  measure->clean2dBoxTable(measure->_met, false);
603 
604  uint idCnt = 1;
605  int ii;
606  for (ii = 0; ii < n - 1; ii++) {
607  measure->createNetSingleWire(_wireDirName, idCnt, w_layout, s_layout);
608  idCnt++;
609  }
610 
611  ii--;
612  int cnt = 0;
613  for (; ii >= 0; ii--) {
614  cnt++;
615  }
616 
617  uint WW = measure->_w_nm;
618  uint SS1;
619  SS1 = measure->_s_nm;
620  uint WW2 = measure->_w2_nm;
621  uint SS2 = measure->_s2_nm;
622 
623  if (n > 1) {
624  cnt++;
625  measure->createNetSingleWire(_wireDirName, idCnt, WW, s_layout);
626  idCnt++;
627 
628  cnt++;
629  if (!measure->_diag) {
630  measure->createNetSingleWire(_wireDirName, idCnt, WW, SS1);
631  idCnt++;
632  cnt++;
633  measure->createNetSingleWire(_wireDirName, idCnt, WW2, SS2);
634  idCnt++;
635  } else {
636  uint met_tmp = measure->_met;
637  measure->_met = measure->_overMet;
638  uint ss2 = SS1;
639  if (measure->_s_nm == 0) {
640  ss2 = measure->_s_nm;
641  }
642 
643  measure->createNetSingleWire(_wireDirName, idCnt, 0, ss2);
644  idCnt++;
645 
646  measure->_met = met_tmp;
647  measure->createNetSingleWire(_wireDirName, idCnt, w_layout, s_layout);
648  idCnt++;
649  }
650 
651  for (int jj = 0; jj < n - 1; jj++) {
652  cnt++;
653  measure->createNetSingleWire(_wireDirName, idCnt, w_layout, s_layout);
654  idCnt++;
655  }
656  } else {
657  cnt++;
658  measure->createNetSingleWire(_wireDirName, 3, w_layout, s_layout);
659  idCnt++;
660  }
661 
662  if (measure->_diag) {
663  return cnt;
664  }
665  bool grid_context = true;
666  bool grid_context_same_dir = false;
667 
668  int cntx_dist = -1; // -1 means min sep
669 
670  int extend_blockage = (measure->_minWidth + measure->_minSpace);
671  int extend_blockage_gap = measure->getPatternExtend();
672 
673  int bboxUR[2]
674  = {measure->_ur[0] + extend_blockage, measure->_ur[1] + extend_blockage};
675  bboxLL[0] -= extend_blockage;
676  bboxLL[1] -= extend_blockage;
677 
678  if (grid_context && (measure->_underMet > 0 || measure->_overMet > 0)) {
679  if (!grid_context_same_dir) {
680  measure->createContextGrid(
681  _wireDirName, bboxLL, bboxUR, measure->_underMet, cntx_dist);
682  measure->createContextGrid(
683  _wireDirName, bboxLL, bboxUR, measure->_overMet, cntx_dist);
684  } else {
685  measure->createContextGrid_dir(
686  _wireDirName, bboxLL, bboxUR, measure->_underMet);
687  measure->createContextGrid_dir(
688  _wireDirName, bboxLL, bboxUR, measure->_overMet);
689  }
690  } else {
691  double pitchMult = 1.0;
692 
693  measure->clean2dBoxTable(measure->_underMet, true);
694  measure->createContextObstruction(_wireDirName,
695  bboxLL[0],
696  bboxLL[1],
697  bboxUR,
698  measure->_underMet,
699  pitchMult);
700 
701  measure->clean2dBoxTable(measure->_overMet, true);
