186 proc 1darray { name data } {
187  upvar $name arr
188  set idx 0
189  set cnt [scan $data "%s %\[^\\]" num rem]
190  while { $cnt > 1 } {
191  if { [string is double $num] } {
192  set arr($idx) $num
193  incr idx
194  }
195  set cnt [scan $rem "%s %\[^\\]" num rem]
196  }
197  if { [string is double $rem] } {
198  set arr($idx) $rem
199  }
200 }

210 proc get2Darray { dir filename SPACE s_name WIDTH w_name A a_name } {
211 
212  upvar 1 $SPACE space
213  upvar 1 $WIDTH width
214  upvar 1 $A 2d_name
215 
216  set datafile [file join $dir $filename ]
217 
218  set fp [open $datafile "r"]
219  set data [read $fp]
220  close $fp
221 
222  set data [split $data "\n"]
223 
224  # read S
225  set i 0
226  set S_flag 0
227  set W_flag 0
228  set dwr_flag 0
229 
230  foreach arg $data {
231  if { [string trim $arg] == $s_name } {
232  set S_flag 1
233  continue
234  }
235  if { [string trim $arg] == $w_name } {
236  set W_flag 1
237  continue
238  }
239  if { [string trim $arg] == $a_name } {
240  set dwr_flag 1
241  continue
242  }
243  if { $S_flag > 0 && [string trim $arg] != "" } {
244  1darray space $arg
245  set S_flag 0
246  continue
247  }
248  if { $W_flag > 0 && [string trim $arg] != "" } {
249  1darray width $arg
250  set W_flag 0
251  continue
252  }
253  if { $dwr_flag>0 && [string trim $arg] != "" } {
254  1darray "a$i" $arg
255  incr i
256  }
257  }
258  if { ($dwr_flag>0) } {
259  for { set j 0 } { $j < $i } {incr j} {
260  2darray 2d_name $j "a$j"
261  }
262  # parray 2d_name
263  }
264 }
