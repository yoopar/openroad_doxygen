138 void extSpef::setupMappingForWrite(uint btermCnt, uint itermCnt)

147  }

161 
162  _btermTable->add(0);
163  _itermTable->add(0);
164 
165  _btermCapTable = new Ath__array1D<double*>(btermCnt);
166  _itermCapTable = new Ath__array1D<double*>(itermCnt);
167  _nodeCapTable = new Ath__array1D<double*>(16000);
168 
169  initCapTable(_btermCapTable);
170  initCapTable(_itermCapTable);
171  initCapTable(_nodeCapTable);

248 void extSpef::reinitCapTable(Ath__array1D<double*>* table, const uint n)
249 {

251  const uint prevCnt = table->getSize();
252  table->resize(n);

254  double* a = new double[_cornerCnt];
255  table->add(a);
256  const uint currentCnt = table->getSize();
257  for (uint ii = prevCnt + 1; ii < currentCnt; ii++) {
258  double* a = new double[_cornerCnt];
259  table->add(a);

263  double* a = table->get(kk);

266  a[jj] = 0.0;

338 void extSpef::writeITerm(const uint node)
339 {
340  odb::dbITerm* iterm = odb::dbITerm::getITerm(_block, node);
341  if (iterm->getInst()->getMaster()->isMarked()) {
342  return;
343  }
344 
345  fprintf(_outFP, "*I ");
346  writeITermNode(node);
347 
348  const char io = iterm->getIoType().getString()[0];
349  fprintf(_outFP, "%c ", io);
350  const int dbunit = _block->getDbUnitsPerMicron();
351  const double db2nm = 1.0 / dbunit;
352  if (_writingNodeCoords == C_ON) {
353  int jid = 0;
354  odb::dbWire* wire = _d_net->getWire();
355  if (wire) {
356  jid = wire->getTermJid(iterm->getId());
357  }
358  odb::Point pt;
359  if (_termJxy && jid) {
360  pt = wire->getCoord(jid);
361  } else {
362  int x1 = 0;
363  int y1 = 0;
364  iterm->getAvgXY(&x1, &y1);
365  pt = {x1, y1};
366  }
367  fprintf(_outFP, "*C %f %f ", db2nm * pt.x(), db2nm * pt.y());
368  }
369  fprintf(_outFP,
370  "*D %s\n",
371  addEscChar(iterm->getMTerm()->getMaster()->getName().c_str(), false));
372 }
