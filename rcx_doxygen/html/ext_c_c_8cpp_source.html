46 uint Ath__track::trackContextOn(int orig,
47  int end,
48  int base,
49  int width,
50  uint firstContextTrack,
51  Ath__array1D<int>* context)
52 {
53  Ath__wire* swire = getTargetWire();
54  if (!swire) {
55  return 0;
56  }
57  while (swire && swire->_xy + swire->_len < base) {
58  swire = nextTargetWire(0); // context of both power/noPower
59  }
60  if (!swire || swire->_xy >= (base + width)
61  || (!firstContextTrack && swire->_srcId)) {
62  return 0;
63  }
64  int p1 = swire->_base > orig ? swire->_base : orig;
65  int send = swire->_base + swire->_width;
66  int p2 = send < end ? send : end;
67  if (p2 <= p1) {
68  return 0;
69  }
70  context->add(p1);
71  context->add(p2);
72  return p2 - p1;
73 }

75 void Ath__grid::gridContextOn(int orig, int len, int base, int width)
76 {
77  Ath__array1D<int>* context = _gridtable->contextArray()[_level];
78  context->resetCnt(0);
79  context->add(orig);
80  int end = orig + len;
81  uint lowTrack = getMinMaxTrackNum(orig);
82  uint hiTrack = getMinMaxTrackNum(orig + len);
83  Ath__track *track, *btrack;
84  uint jj;
85  uint firstContextTrack = 1;
86  bool tohi = _gridtable->targetHighTracks() > 0 ? true : false;
87  for (jj = lowTrack; jj <= hiTrack; jj++) {
88  btrack = _trackTable[jj];
89  if (btrack == nullptr) {
90  continue;
91  }
92  track = nullptr;
93  while ((track = btrack->getNextSubTrack(track, tohi))) {
94  track->trackContextOn(orig, end, base, width, firstContextTrack, context);
95  firstContextTrack = 0;
96  }
97  }
98  context->add(end);
99 }

119 uint Ath__track::findOverlap(Ath__wire* origWire,
120  uint ccThreshold,
121  Ath__array1D<Ath__wire*>* wTable,
122  Ath__array1D<Ath__wire*>* nwTable,
123  Ath__grid* ccGrid,
124  Ath__array1D<Ath__wire*>* ccTable,
125  uint met,
126  rcx::CoupleAndCompute coupleAndCompute,
127  void* compPtr)
128 {
129  rcx::CoupleOptions coupleOptions{};
130 
131  AthPool<Ath__wire>* wirePool = _grid->getWirePoolPtr();
132 
133  uint NoPowerTarget = _grid->getGridTable()->noPowerTarget();
134  bool srcMarked = origWire->getNet()->isMarked();
135  uint TargetHighMarkedNet = _grid->getGridTable()->targetHighMarkedNet();
136  bool allNet = _grid->getGridTable()->allNet();
137  bool needMarkedNetW2 = !allNet && TargetHighMarkedNet && !srcMarked;
138  int TTTnoInNetCC = 0;
139  int len1, len2, len3, rc;
140  Ath__wire* w2 = getTargetWire();
141  bool targetHiTrack
142  = _grid->getGridTable()->targetHighTracks() > 0 ? true : false;
143  bool targetReversed = _grid->getGridTable()->targetTrackReversed();
144  bool useDbSdb = _grid->getGridTable()->usingDbSdb();
145 
146  uint last = wTable->getCnt();
147  bool inThreshold, skipCCgen;
148  bool notExtractedW2;
149  int exid;
150  for (uint ii = 0; ii < last; ii++) {
151  Ath__wire* w1 = wTable->get(ii);
152 
153  if (w2 == nullptr) {
154  nwTable->add(w1);
155  continue;
156  }
157 
158  while (true) {
159  notExtractedW2
160  = useDbSdb && !w2->isPower() && !w2->isVia()
161  && w2->getNet()->getWire()
162  && (!w2->getNet()->getWire()->getProperty((int) w2->_otherId, exid)
163  || exid == 0);
164  len1 = 0;
165  len2 = 0;
166  len3 = 0;
167 
168  inThreshold = true;
169  if ((targetHiTrack
170  && w2->_base >= w1->_base + (int) (w1->_width + ccThreshold))
171  || (!targetHiTrack
172  && w1->_base >= w2->_base + (int) (w2->_width + ccThreshold))) {
173  inThreshold = false;
174  }
175 
176  skipCCgen = _grid->getGridTable()->handleEmptyOnly()
177  || (needMarkedNetW2 && !w2->getNet()->isMarked());
178  rc = w1->wireOverlap(w2, &len1, &len2, &len3);
179 
180  if (inThreshold == true && rc == 0) {
181  if (len1 > 0) {
182  // create empty wire and ADD on emptyTable!
183  Ath__wire* newEmptyWire = origWire->makeWire(wirePool, w1->_xy, len1);
184  nwTable->add(newEmptyWire);
185  }
186  Ath__wire* wtwo = w2->_srcId ? _grid->getWirePtr(w2->_srcId) : w2;
187  // create cc and ADD on ccTable
188  if (!notExtractedW2 && len2 > 0 && !skipCCgen
189  && (!TTTnoInNetCC || wtwo->isPower() || origWire->isPower()
190  || origWire->getNet() != wtwo->getNet())
191  && (!targetReversed || wtwo->isPower()
192  || !wtwo->getNet()->isMarked())) {
193  if (coupleAndCompute == nullptr) {
194  Ath__wire* ovWire
195  = origWire->makeCoupleWire(ccGrid->getWirePoolPtr(),
196  targetHiTrack,
197  wtwo,
198  w1->_xy + len1,
199  len2,
200  ccGrid->defaultWireType());
201 
202  if (ovWire == nullptr) {
203  _grid->getGridTable()->incrCCshorts();
204  } else {
205  ccTable->add(ovWire);
206  ccGrid->placeWire(ovWire);
207  ovWire->_flags = ccGrid->defaultWireType();
208  }
209  } else {
210  Ath__wire* topwire = targetHiTrack ? wtwo : origWire;
211  Ath__wire* botwire = targetHiTrack ? origWire : wtwo;
212  int dist = topwire->_base - (botwire->_base + botwire->_width);
213 
214  if (dist > 0) {
215  _grid->contextsOn(
216  w1->_xy + len1, len2, botwire->_base + botwire->_width, dist);
217  coupleOptions[0] = met;
218 
219  int bBoxId = (int) botwire->_boxId;
220  if (useDbSdb) {
221  bBoxId = botwire->getRsegId();
222  }
223 
224  coupleOptions[1] = bBoxId; // dbRSeg id for SRC segment
225 
226  if (botwire->_otherId == 0) {
227  coupleOptions[1] = -bBoxId; // POwer SBox Id
228  }
229 
230  int tBoxId = (int) topwire->_boxId;
231 
232  if (useDbSdb) {
233  tBoxId = topwire->getRsegId();
234  }
235 
236  coupleOptions[2] = tBoxId; // dbRSeg id for TARGET segment
237  if (topwire->_otherId == 0) {
238  coupleOptions[2] = -tBoxId; // POwer SBox Id
239  }
240 
241  coupleOptions[3] = len2;
242  coupleOptions[4] = dist;
243  coupleOptions[5] = w1->_xy + len1;
244  coupleOptions[6] = botwire->_dir;
245 
246  coupleOptions[7] = botwire->_width;
247  coupleOptions[8] = topwire->_width;
248  coupleOptions[9] = botwire->_base;
249  coupleOptions[11] = targetHiTrack ? 1 : 0;
250  coupleOptions[18] = botwire->_track->getTrackNum();
251  coupleAndCompute(coupleOptions, compPtr);
252  }
253  }
254  }
255 
256  if (len3 > 0) {
257  w1->setXY(w1->_xy + len1 + len2, len3);
258  w2 = nextTargetWire(NoPowerTarget);
259  if (!w2) {
260  nwTable->add(w1);
261  break;
262  }
263  continue;
264  }
265  wirePool->free(w1);
266  break;
267  }
268  if (inThreshold == false && rc == 0) {
269  if (len3 > 0) {
270  w2 = nextTargetWire(NoPowerTarget);
271  if (!w2) {
272  nwTable->add(w1);
273  break;
274  }
275  continue;
276  }
277  nwTable->add(w1);
278  break;
279  }
280  if (rc == 1) {
281  w2 = nextTargetWire(NoPowerTarget);
282  if (!w2) {
283  nwTable->add(w1);
284  break;
285  }
286  continue;
287  }
288  nwTable->add(w1);
289  break;
290  }
291  }
292  return nwTable->getCnt();
293 }

309 Ath__track* Ath__track::nextTrackInRange(uint& delt,
310  uint trackDist,
311  uint srcTrack,
312  bool tohi)
313 {
314  Ath__track* ttrack = nullptr;
315  uint tgtTnum;
316  while (ttrack == nullptr) {
317  delt++;
318  if (delt > trackDist || (!tohi && srcTrack < delt)) {
319  return nullptr;
320  }

322  if (tgtTnum >= _grid->getTrackCnt()) {
323  return nullptr;
324  }
325  ttrack = _grid->getTrackPtr(tgtTnum);
326  }

348 uint Ath__track::couplingCaps(Ath__grid* ccGrid,
349  uint srcTrack,
350  uint trackDist,
351  uint ccThreshold,
352  Ath__array1D<uint>* ccIdTable,
353  uint met,
354  CoupleAndCompute coupleAndCompute,
355  void* compPtr)
356 {
357  Ath__track* tstrack;
358  bool tohi = _grid->getGridTable()->targetHighTracks() > 0 ? true : false;
359  initTargetTracks(srcTrack, trackDist, tohi);
360 
361  uint dir = _grid->getDir();
362  rcx::CoupleOptions coupleOptions{};
363 
364  Ath__array1D<Ath__wire*> w1Table;
365  Ath__array1D<Ath__wire*> w2Table;
366  Ath__array1D<Ath__wire*>*wTable, *nwTable, *twTable;
367  Ath__array1D<Ath__wire*> ccTable;
368 
369  bool useDbSdb = _grid->getGridTable()->usingDbSdb();
370  int noPowerSource = _grid->getGridTable()->noPowerSource();
371  uint TargetHighMarkedNet = _grid->getGridTable()->targetHighMarkedNet();
372  bool allNet = _grid->getGridTable()->allNet();
373  AthPool<Ath__wire>* wirePool = _grid->getWirePoolPtr();
374  uint wireCnt = 0;
375  Ath__wire* origWire = nullptr;
376  uint delt;
377  int exid;
378 
379  if (ttttGetDgOverlap) {
380  // to initTargetSeq
381  coupleOptions[0] = -met;
382  coupleOptions[5] = 1;
383  coupleAndCompute(coupleOptions, compPtr);
384  }
385  int nexy, nelen;
386  Ath__wire* wire = nullptr;
387  Ath__wire* pwire = nullptr;
388  Ath__wire* nwire = getNextWire(wire);
389  for (wire = nwire; wire; pwire = wire, wire = nwire) {
390  nwire = getNextWire(wire);
391  if (!wire->isPower() && nwire && nwire->isPower()
392  && nwire->_xy < wire->_xy + wire->_len) {
393  coupleOptions[19] = 0; // bp
394  }
395  if (wire->isPower() && nwire && !nwire->isPower()
396  && nwire->_xy < wire->_xy + wire->_len) {
397  coupleOptions[19] = 0; // bp
398  }
399 
400  if (noPowerSource && wire->isPower()) {
401  continue;
402  }
403 
404  // when !TargetHighMarkedNet, need only marked source
405  if (!allNet && !TargetHighMarkedNet && !wire->getNet()->isMarked()) {
406  continue;
407  }
408  if (tohi
409  && _grid->getMinMaxTrackNum(wire->_base + wire->_width) != srcTrack) {
410  continue;
411  }
412  if (!tohi && wire->_srcId > 0) {
413  continue;
414  }
415  if (useDbSdb && !wire->isPower() && !wire->isVia()
416  && wire->getNet()->getWire()
417  && (!wire->getNet()->getWire()->getProperty((int) wire->_otherId, exid)
418  || exid == 0)) {
419  continue;
420  }
421 
422  wireCnt++;
423 
424  w1Table.resetCnt();
425  wTable = &w1Table;
426  nwTable = &w2Table;
427  nexy = wire->_xy;
428  nelen = wire->_len;
429  int delta = 0;
430  if (pwire) {
431  delta = pwire->_xy + pwire->_len - wire->_xy;
432  }
433  if (pwire && delta > 0
434  && wire->_base + wire->_width < pwire->_base + pwire->_width) {
435  nexy += delta;
436  nelen -= delta;
437  }
438  if (nwire) {
439  delta = wire->_xy + wire->_len - nwire->_xy;
440  }
441  if (nwire && delta > 0
442  && wire->_base + wire->_width < nwire->_base + nwire->_width) {
443  nelen -= delta;
444  }
445  if (nelen <= 0) { // or nelen < wire->_width
446  continue;
447  }
448  Ath__wire* newEmptyWire = wire->makeWire(wirePool, nexy, nelen);
449  wTable->add(newEmptyWire);
450 
451  delt = 0;
452  tstrack = this;
453  while (nextSubTrackInRange(tstrack, delt, trackDist, srcTrack, tohi)) {
454  nwTable->resetCnt();
455 
456  origWire = wire->_srcId ? _grid->getWirePtr(wire->_srcId) : wire;
457  tstrack->findOverlap(origWire,
458  ccThreshold,
459  wTable,
460  nwTable,
461  ccGrid,
462  &ccTable,
463  met,
464  coupleAndCompute,
465  compPtr);
466 
467  twTable = wTable;
468  wTable = nwTable;
469  nwTable = twTable;
470 
471  if (wTable->getCnt() == 0) {
472  break;
473  }
474  }
475  if (coupleAndCompute != nullptr) {
476  bool visited = false;
477  int wBoxId = 0;
478 
479  for (uint kk = 0; kk < wTable->getCnt(); kk++) {
480  Ath__wire* empty = wTable->get(kk);
481 
482  coupleOptions[0] = met;
483 
484  wBoxId = (int) wire->_boxId;
485  if (useDbSdb) {
486  wBoxId = wire->getRsegId();
487  }
488 
489  coupleOptions[1] = wBoxId; // dbRSeg id
490  if (wire->_otherId == 0) {
491  coupleOptions[1] = -wBoxId; // dbRSeg id
492  }
493 
494  coupleOptions[2] = 0; // dbRSeg id
495 
496  coupleOptions[3] = empty->_len;
497  coupleOptions[4] = -1;
498  coupleOptions[5] = empty->_xy;
499  coupleOptions[6] = wire->_dir;
500 
501  coupleOptions[7] = wire->_width;
502  coupleOptions[8] = 0;
503  coupleOptions[9] = wire->_base;
504  coupleOptions[10] = dir;
505 
506  coupleOptions[11] = tohi ? 1 : 0;
507 
508  bool ignore_visited = true;
509  if (ignore_visited || (wire->_visited == 0 && wire->_srcWire == nullptr)
510  || (wire->_srcWire != nullptr && wire->_srcWire->_visited == 0)) {
511  if (wire->_srcWire != nullptr) {
512  coupleOptions[20] = wire->_srcWire->_ouLen;
513  } else {
514  coupleOptions[20] = wire->_ouLen;
515  }
516  coupleAndCompute(coupleOptions, compPtr);
517  visited = true;
518  if (wire->_srcWire != nullptr) {
519  wire->_srcWire->_ouLen = coupleOptions[20];
520  } else {
521  wire->_ouLen = coupleOptions[20];
522  }
523  }
524  wirePool->free(empty);
525  }
526  if (visited) {
527  wire->_visited = 1;
528  }
529  }
530  }
531  if (coupleAndCompute == nullptr) {
532  for (uint kk = 0; kk < ccTable.getCnt(); kk++) {
533  Ath__wire* v = ccTable.get(kk);
534  ccIdTable->add(v->_id);
535  }
536  }
537  return wireCnt;
538 }

641 void Ath__track::buildDgContext(Ath__array1D<SEQ*>* dgContext,
642  Ath__wire**& allWire,
643  int& awcnt,
644  int& awsize)
645 {
646  std::vector<Ath__wire*> ctxwire;
647  Ath__track* track = nullptr;
648  bool tohi = true;
649  uint tcnt = 0;
650  while ((track = getNextSubTrack(track, tohi))) {
651  tcnt++;
652  track->getTrackWires(ctxwire);
653  }
654  uint ctxsize = ctxwire.size();
655  if (ctxsize == 0) {
656  return;
657  }
658  if (tcnt > 1 && ctxsize > 1) {
659  std::sort(ctxwire.begin(), ctxwire.end(), compareAthWire());
660  }
661  uint jj;
662  Ath__wire* nwire;
663  uint xidx = 0;
664  uint yidx = 1;
665  uint lidx, bidx;
666  AthPool<SEQ>* seqPool = _grid->getGridTable()->seqPool();
667  SEQ* seq;
668  int rsegid;
669  for (jj = 0; jj < ctxsize; jj++) {
670  if (awcnt == awsize) {
671  awsize += 1024;
672  allWire = (Ath__wire**) realloc(allWire, sizeof(Ath__wire*) * awsize);
673  }
674  nwire = ctxwire[jj];
675  allWire[awcnt++] = nwire;
676  seq = seqPool->alloc();
677  lidx = _grid->getDir() == 1 ? xidx : yidx;
678  bidx = _grid->getDir() == 1 ? yidx : xidx;
679  seq->_ll[lidx] = nwire->_xy;
680  seq->_ll[bidx] = nwire->_base;
681  seq->_ur[lidx] = nwire->_xy + nwire->_len;
682  seq->_ur[bidx] = nwire->_base + nwire->_width;
683  if (nwire->isPower()) {
684  seq->type = 0;
