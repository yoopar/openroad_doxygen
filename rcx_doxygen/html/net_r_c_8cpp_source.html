205 void extMain::getViaCapacitance(dbShape svia, dbNet* net)
206 {
207  bool USE_DB_UNITS = false;
208 
209  std::vector<dbShape> shapes;
210  dbShape::getViaBoxes(svia, shapes);
211 
212  int Width[32];
213  int Len[32];
214  int Level[32];
215  for (uint jj = 1; jj < 32; jj++) {
216  Level[jj] = 0;
217  Width[jj] = 0;
218  Len[jj] = 0;
219  }
220 
221  std::vector<dbShape>::iterator shape_itr;
222  for (shape_itr = shapes.begin(); shape_itr != shapes.end(); ++shape_itr) {
223  dbShape s = *shape_itr;
224 
225  if (s.getTechLayer()->getType() == dbTechLayerType::CUT) {
226  continue;
227  }
228 
229  int x1 = s.xMin();
230  int y1 = s.yMin();
231  int x2 = s.xMax();
232  int y2 = s.yMax();
233  int dx = x2 - x1;
234  int dy = y2 - y1;
235  int width = std::min(dx, dy);
236  int len = std::max(dx, dy);
237 
238  uint level = s.getTechLayer()->getRoutingLevel();
239  if (Len[level] < len) {
240  Len[level] = len;
241  Width[level] = width;
242  Level[level] = level;
243  }
244  if (USE_DB_UNITS) {
245  width = GetDBcoords2(width);
246  len = GetDBcoords2(len);
247  }
248 
249  if (net->getId() == _debug_net_id) {
250  debugPrint(
251  logger_,
252  RCX,
253  "extrules",
254  1,
255  "VIA_CAP: C getViaCapacitance: {} {} {} {} M{} W {} LEN {} n{}",
256  x1,
257  x2,
258  y1,
259  y2,
260  level,
261  width,
262  len,
263  _debug_net_id);
264  }
265  }
266  for (uint jj = 1; jj < 32; jj++) {
267  if (Level[jj] == 0) {
268  continue;
269  }
270 
271  int w = Width[jj];
272  int len = Len[jj];
273 
274  if (USE_DB_UNITS) {
275  w = GetDBcoords2(w);
276  len = GetDBcoords2(len);
277  }
278  for (uint ii = 0; ii < _metRCTable.getCnt(); ii++) {
279  double areaCap;
280  double c1 = getFringe(jj, w, ii, areaCap);
281 
282  _tmpCapTable[ii] = len * 2 * c1;
283  if (net->getId() == _debug_net_id) {
284  debugPrint(logger_,
285  RCX,
286  "extrules",
287  1,
288  "VIA_CAP: C getViaCapacitance: M{} W {} LEN {} eC={:.3f} "
289  "tC={:.3f} {} n{}",
290  jj,
291  w,
292  len,
293  c1,
294  _tmpCapTable[ii],
295  ii,
296  _debug_net_id);
297  }
298  }
299  }

302 void extMain::getShapeRC(dbNet* net,
303  const dbShape& s,
304  const Point& prevPoint,
305  const dbWirePathShape& pshape)
306 {
307  if (s.isVia()) {
308  double res = 0.0;
309  uint level = 0;
310  if (dbTechVia* tvia = s.getTechVia()) {
311  level = tvia->getBottomLayer()->getRoutingLevel();
312  res = tvia->getResistance();
313  if (res == 0) {
314  res = getViaResistance(tvia);
315  }
316  if (res > 0) {
317  tvia->setResistance(res);
318  }
319  if (res <= 0.0) {
320  const uint width = tvia->getBottomLayer()->getWidth();
321  res = getResistance(level, width, width, 0);
322  }
323  } else {
324  dbVia* bvia = s.getVia();
325  level = bvia->getBottomLayer()->getRoutingLevel();
326  res = getViaResistance_b(bvia, net);
327 
328  if (res <= 0.0) {
329  const uint width = bvia->getBottomLayer()->getWidth();
330  res = getResistance(level, width, width, 0);
331  }
332  }
333  if (level > 0) {
334  getViaCapacitance(s, net);
335  for (uint ii = 0; ii < _metRCTable.getCnt(); ii++) {
336  _tmpResTable[ii] = res;
337  }
338  }
339  } else {
340  const uint level = s.getTechLayer()->getRoutingLevel();
341  const auto [width, len]
342  = std::minmax({pshape.shape.getDX(), pshape.shape.getDY()});
343 
344  if (_lef_res) {
345  const double res = getResistance(level, width, len, 0);
346  _tmpResTable[0] = res;
347  } else {
348  for (uint ii = 0; ii < _metRCTable.getCnt(); ii++) {
349  double areaCap;
350  getFringe(level, width, ii, areaCap);
351 
352  _tmpCapTable[ii] = 0;
353  const double r = getResistance(level, width, len, ii);
354  _tmpResTable[ii] = r;
355  }
356  }
357  if (!_allNet && _couplingFlag > 0) {
358  _ccMinX = std::min(s.xMin(), _ccMinX);
359  _ccMinY = std::min(s.yMin(), _ccMinY);
360  _ccMaxX = std::max(s.xMax(), _ccMaxX);
361  _ccMaxY = std::max(s.yMax(), _ccMaxY);
362  }
363  }
