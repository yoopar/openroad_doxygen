227 uint extMain::getResCapTable()
228 {
229  calcMinMaxRC();
230  _currentModel = getRCmodel(0);
231 
232  extMeasure m(logger_);
233  m._underMet = 0;
234  m._overMet = 0;
235 
236  uint cnt = 0;
237  for (dbTechLayer* layer : _tech->getLayers()) {
238  if (layer->getRoutingLevel() == 0) {
239  continue;
240  }
241 
242  const uint n = layer->getRoutingLevel();
243 
244  const uint w = layer->getWidth(); // nm
245  _minWidthTable[n] = w;
246 
247  m._width = w;
248  m._met = n;
249 
250  uint sp = layer->getSpacing(); // nm
251  _minDistTable[n] = sp;
252  if (sp == 0) {
253  sp = layer->getPitch() - layer->getWidth();
254  _minDistTable[n] = sp;
255  }
256  double resTable[20];
257  for (uint jj = 0; jj < _modelMap.getCnt(); jj++) {
258  resTable[jj] = 0.0;
259  }
260  calcRes0(resTable, n, w, 1);
261  for (uint jj = 0; jj < _modelMap.getCnt(); jj++) {
262  const uint modelIndex = _modelMap.get(jj);
263  extMetRCTable* rcModel = _currentModel->getMetRCTable(modelIndex);
264 
265  const double res = layer->getResistance(); // OHMS per square
266  _resistanceTable[jj][n] = res;
267 
268  _capacitanceTable[jj][n] = 0.0;
269 
270  extDistRC* rc = rcModel->getOverFringeRC(&m);
271 
272  if (rc != nullptr) {
273  const double r1 = rc->getRes();
274  _capacitanceTable[jj][n] = rc->getFringe();
275  debugPrint(logger_,
276  RCX,
277  "extrules",
278  1,
279  "EXT_RES: R "
280  "Layer= {} met= {} w= {} cc= {:g} fr= {:g} res= {:g} "
281  "model_res= {:g} new_model_res= {:g}",
282  layer->getConstName(),
283  n,
284  w,
285  rc->getCoupling(),
286  rc->getFringe(),
287  res,
288  r1,
289  resTable[jj]);
290  }
291 
292  if (!_lef_res) {
293  _resistanceTable[jj][n] = resTable[jj];
294  } else {
295  debugPrint(logger_,
296  RCX,
297  "extrules",
298  1,
299  "EXT_RES_LEF: R Layer= {} met= {} lef_res= {:g}",
300  layer->getConstName(),
301  n,
302  res);
303  }
304  }
305  cnt++;
306  }
307  return cnt;
308 }

310 bool extMain::checkLayerResistance()
311 {
312  dbSet<dbTechLayer> layers = _tech->getLayers();
313  dbSet<dbTechLayer>::iterator itr;
314 
315  uint cnt = 0;
316  for (itr = layers.begin(); itr != layers.end(); ++itr) {
317  dbTechLayer* layer = *itr;
318 
319  if (layer->getRoutingLevel() == 0) {
320  continue;
321  }
322 
323  double res = layer->getResistance(); // OHMS per square
324 
325  if (res <= 0.0) {
326  logger_->warn(RCX,
327  139,
328  "Missing Resistance value for layer {}",
329  layer->getConstName());
330  cnt++;
331  }
332  }
333  if (cnt > 0) {
334  logger_->warn(RCX,
335  138,
336  "{} layers are missing resistance value; Check LEF file. "
337  "Extraction cannot proceed! Exiting",
338  cnt);
339  return false;
340  }
341  return true;

482 void extMain::updateTotalCap(dbRSeg* rseg,
483  extMeasure* m,
484  const double* deltaFr,
485  uint modelCnt,
486  bool includeCoupling,
487  bool includeDiag)
488 {
489  double tot, cap;
490  int extDbIndex, sci, scDbIdx;
491  for (uint modelIndex = 0; modelIndex < modelCnt; modelIndex++) {
492  extDistRC* rc = m->_rc[modelIndex];
493 
494  double frCap = rc->_fringe;
495 
496  double ccCap = 0.0;
497  if (includeCoupling) {
498  ccCap = rc->_coupling;
499  }
500 
501  double diagCap = 0.0;
502  if (includeDiag) {
503  diagCap = rc->_diag;
504  }
505 
506  cap = frCap + ccCap + diagCap - deltaFr[modelIndex];
507  if (_gndcModify) {
508  cap *= _gndcFactor;
509  }
510 
511  extDbIndex = getProcessCornerDbIndex(modelIndex);
512  tot = rseg->getCapacitance(extDbIndex);
513  tot += cap;
514 
515  rseg->setCapacitance(tot, extDbIndex);
516  getScaledCornerDbIndex(modelIndex, sci, scDbIdx);
517  if (sci == -1) {
518  continue;
519  }
520  getScaledGndC(sci, cap);
521  tot = rseg->getCapacitance(scDbIdx);
522  tot += cap;
523  rseg->setCapacitance(tot, scDbIdx);
524  }
525 }

570 void extMain::measureRC(CoupleOptions& options)
571 {
572  _totSegCnt++;
573  int rsegId1 = options[1]; // dbRSeg id for SRC segment
574  int rsegId2 = options[2]; // dbRSeg id for Target segment
575 
576  if ((rsegId1 < 0) && (rsegId2 < 0)) { // power nets
577  return;
578  }
579 
580  extMeasure m(logger_);
581  m.defineBox(options);
582  m._ccContextArray = _ccContextArray;
583  m._ccMergedContextArray = _ccMergedContextArray;
584 
585  dbRSeg* rseg1 = nullptr;
586  dbNet* srcNet = nullptr;
587  if (rsegId1 > 0) {
588  rseg1 = dbRSeg::getRSeg(_block, rsegId1);
589  srcNet = rseg1->getNet();
590  printNet(srcNet, _debug_net_id);
591  }
592 
593  dbRSeg* rseg2 = nullptr;
594  dbNet* tgtNet = nullptr;
595  if (rsegId2 > 0) {
596  rseg2 = dbRSeg::getRSeg(_block, rsegId2);
597  tgtNet = rseg2->getNet();
598  printNet(tgtNet, _debug_net_id);
599  }
600 
601  m._pixelTable = _geomSeq;
602 
603  _totSignalSegCnt++;
604 
605  _currentModel = getRCmodel(0);
606  m._currentModel = _currentModel;
607  for (uint ii = 0; ii < _metRCTable.getCnt(); ii++) {
608  m._metRCTable.add(_metRCTable.get(ii));
609  }
610  if (m._met >= _currentModel->getLayerCnt()) { // TO_TEST
611  return;
612  }
613 
614  m._layerCnt = _currentModel->getLayerCnt();
615 
616  double deltaFr[20];
617  for (uint jj = 0; jj < m._metRCTable.getCnt(); jj++) {
618  deltaFr[jj] = 0.0;
619  m._rc[jj]->_coupling = 0.0;
620  m._rc[jj]->_fringe = 0.0;
621  m._rc[jj]->_diag = 0.0;
622  m._rc[jj]->_res = 0.0;
623  m._rc[jj]->_sep = 0;
624  }
625 
626  uint totLenCovered = 0;
627  if (_usingMetalPlanes) {
628  if (_ccContextArray
629  && ((!srcNet || (int) srcNet->getId() == ttttsrcnet)
630  || (!tgtNet || (int) tgtNet->getId() == tttttgtnet))
631  && (!ttttm || m._met == ttttm)) {
632  int pxy = m._dir ? m._ll[0] : m._ll[1];
633  int pbase = m._dir ? m._ur[1] : m._ur[0];
634  logger_->info(RCX,
635  141,
636  "Context of layer {} xy={} len={} base={} width={}",
637  m._met,
638  pxy,
639  m._len,
640  pbase,
641  m._s_nm);
642  uint ii, jj;
643  for (ii = 1; ii <= _ccContextDepth
644  && (int) ii + m._met < _currentModel->getLayerCnt();
645  ii++) {
646  logger_->info(RCX, 142, " layer {}", ii + m._met);
647  for (jj = 0; jj < _ccContextArray[ii + m._met]->getCnt(); jj++) {
648  logger_->info(RCX,
649  476,
650  " {}: {}",
651  jj,
652  _ccContextArray[ii + m._met]->get(jj));
653  }
654  }
655  for (ii = 1; ii <= _ccContextDepth && m._met - ii > 0; ii++) {
656  logger_->info(RCX, 65, " layer {}", m._met - ii);
657  for (jj = 0; jj < _ccContextArray[m._met - ii]->getCnt(); jj++) {
658  logger_->info(RCX,
659  143,
660  " {}: {}",
661  jj,
662  _ccContextArray[m._met - ii]->get(jj));
663  }
664  }
665  }
666  totLenCovered = m.measureOverUnderCap();
667  }
668  int lenOverSub = m._len - totLenCovered;
669 
670  if (m._dist < 0) { // dist is infinit
671 
672  if (totLenCovered > 0) {
673  m._underMet = 0;
674  for (uint jj = 0; jj < m._metRCTable.getCnt(); jj++) {
675  extDistRC* rc = m._metRCTable.get(jj)->getOverFringeRC(&m);
676  deltaFr[jj] = rc->getFringe() * totLenCovered;
677  }
678  }
679  } else { // dist based
680 
681  if (lenOverSub > 0) {
682  m._underMet = 0;
683  m.computeOverRC(lenOverSub);
684  }
685  m.getFringe(m._len, deltaFr);
686 
687  if ((rsegId1 > 0) && (rsegId2 > 0)) { // signal nets
688 
689  _totCCcnt++; // TO_TEST
690 
691  if (m._rc[_minModelIndex]->_coupling < _coupleThreshold) { // TO_TEST
692  updateTotalCap(rseg1, &m, deltaFr, m._metRCTable.getCnt(), true);
693  updateTotalCap(rseg2, &m, deltaFr, m._metRCTable.getCnt(), true);
694 
695  _totSmallCCcnt++;
696 
697  return;
698  }
699  _totBigCCcnt++;
700 
701  dbCCSeg* ccap = dbCCSeg::create(
702  dbCapNode::getCapNode(_block, rseg1->getTargetNode()),
703  dbCapNode::getCapNode(_block, rseg2->getTargetNode()),
704  true);
705 
706  int extDbIndex, sci, scDbIdx;
707  for (uint jj = 0; jj < m._metRCTable.getCnt(); jj++) {
708  extDbIndex = getProcessCornerDbIndex(jj);
709  ccap->addCapacitance(m._rc[jj]->_coupling, extDbIndex);
710  getScaledCornerDbIndex(jj, sci, scDbIdx);
