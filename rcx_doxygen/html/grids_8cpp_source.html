457 int Ath__wire::wireOverlap(Ath__wire* w, int* len1, int* len2, int* len3)
458 {
459  // _xy, _len : reference rect
460 
461  int X1 = _xy;
462  int DX = _len;
463  int x1 = w->_xy;
464  int dx = w->_len;
465 
466  // fprintf(stdout, "%d %d %d %d : ", X1, DX, x1, dx);
467 
468  int dx1 = X1 - x1;
469  //*len1= dx1;
470  if (dx1 >= 0) // on left side
471  {
472  int dlen = dx - dx1;
473  if (dlen <= 0) {
474  return 1;
475  }
476 
477  *len1 = 0;
478  int DX2 = dlen - DX;
479 
480  if (DX2 <= 0) {
481  *len2 = dlen;
482  *len3 = -DX2;
483  } else {
484  *len2 = DX;
485  //*len3= DX2;
486  *len3 = 0;
487  }
488  } else {
489  *len1 = -dx1;
490 
491  if (dx1 + DX <= 0) { // outside right side
492  return 2;
493  }
494 
495  int DX2 = (x1 + dx) - (X1 + DX);
496  if (DX2 > 0) {
497  *len2 = DX + dx1; // dx1 is negative
498  *len3 = 0;
499  } else {
500  *len2 = dx;
501  *len3 = -DX2;
502  }
503  }
504  return 0;
505 }
