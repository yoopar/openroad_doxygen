77 void Ext::bench_wires(const BenchWiresOptions& bwo)
78 {
79  extMainOptions opt;
80 
81  opt._topDir = bwo.dir;
82  opt._met_cnt = bwo.met_cnt;
83  opt._met = bwo.met;
84  opt._overDist = bwo.over_dist;
85  opt._underDist = bwo.under_dist;
86  opt._overMet = bwo.over_met;
87  opt._over = bwo.Over;
88  opt._underMet = bwo.under_met;
89  opt._overUnder = bwo.over_under;
90  opt._len = 1000 * bwo.len;
91  opt._wireCnt = bwo.cnt;
92  opt._name = bwo.block;
93 
94  opt._default_lef_rules = bwo.default_lef_rules;
95  opt._nondefault_lef_rules = bwo.nondefault_lef_rules;
96 
97  opt._multiple_widths = bwo.multiple_widths;
98 
99  opt._write_to_solver = bwo.write_to_solver;
100  opt._read_from_solver = bwo.read_from_solver;
101  opt._run_solver = bwo.run_solver;
102  opt._diag = bwo.diag;
103  opt._db_only = bwo.db_only;
104  opt._gen_def_patterns = bwo.gen_def_patterns;
105 
106  opt._res_patterns = bwo.resPatterns;
107 
108  if (opt._gen_def_patterns) {
109  opt._diag = true;
110  opt._overUnder = true;
111  opt._db_only = true;
112  opt._over = true;
113  opt._underMet = 0;
114  }
115 
116  Ath__parser parser(logger_);
117 
118  std::string th_list(bwo.th_list);
119  std::string w_list(bwo.w_list);
120  std::string s_list(bwo.s_list);
121  std::string th(bwo.th);
122  std::string w(bwo.w);
123  std::string s(bwo.s);
124  std::string d(bwo.d);
125  std::string grid_list(bwo.grid_list);
126 
127  if (!th.empty()) {
128  parser.mkWords(bwo.th_list);
129  parser.getDoubleArray(&opt._thicknessTable, 0);
130  opt._thListFlag = true;
131  } else {
132  parser.mkWords(bwo.th);
133  parser.getDoubleArray(&opt._thicknessTable, 0);
134  opt._thListFlag = false;
135  }
136  opt._listsFlag = false;
137  opt._wsListFlag = false;
138  if (opt._default_lef_rules) { // minWidth, minSpacing, minThickness, pitch
139  // multiplied by grid_list
140  // user sets default_flag and grid_list multipliers
141  if (grid_list.empty()) {
142  opt._gridTable.add(1.0);
143  } else {
144  parser.mkWords(grid_list.c_str());
145  parser.getDoubleArray(&opt._gridTable, 0);
146  }
147  opt._listsFlag = true;
148  } else if (opt._nondefault_lef_rules) {
149  opt._listsFlag = true;
150  } else if (!w_list.empty() && !s_list.empty()) {
151  opt._listsFlag = true;
152  opt._wsListFlag = true;
153  parser.mkWords(w_list.c_str());
154  parser.getDoubleArray(&opt._widthTable, 0);
155  parser.mkWords(s_list.c_str());
156  parser.getDoubleArray(&opt._spaceTable, 0);
157  } else {
158  parser.mkWords(w.c_str());
159  parser.getDoubleArray(&opt._widthTable, 0);
160  parser.mkWords(s.c_str());
161  parser.getDoubleArray(&opt._spaceTable, 0);
162  parser.mkWords(th.c_str());
163  parser.getDoubleArray(&opt._thicknessTable, 0);
164  parser.mkWords(d.c_str());
165  parser.getDoubleArray(&opt._densityTable, 0);
166  }
167  _ext->benchWires(&opt);
168 }
