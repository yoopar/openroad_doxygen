132  unsigned int l_find_largest_prime_below_number(unsigned int number)
133  {
134  assert(number > 3);
135  int i = 0;
136  while (m_listOfPrimes[i] > number) {
137  i++;
138  }
139  return m_listOfPrimes[i];
140  }

142  unsigned int hashFunction(const char* key, unsigned int, unsigned int prime)
143  {
144  unsigned int hash = 0;
145  int c;
146 
147  while ((c = static_cast<unsigned char>(*key++)) != '\0') {
148  hash = c + (hash << 6) + (hash << 16) - hash;
149  }
150 
151  return hash % prime;
152  }
