62 uint extMain::getBucketNum(int base, int max, uint step, int xy)
63 {
64  if (xy >= max) {
65  xy = max - 1;

67 
68  int delta = xy - base;

71  }
72 
73  uint n = delta / step;
74  return n;

287 uint extMain::initSearchForNets(int* X1,
288  int* Y1,
289  uint* pitchTable,
290  uint* widthTable,
291  uint* dirTable,
292  Rect& extRect,
293  bool skipBaseCalc)
294 {
295  bool USE_DB_UNITS = false;
296  uint W[32];
297  uint S[32];
298 
299  dbSet<dbTechLayer> layers = _tech->getLayers();
300  dbSet<dbTechLayer>::iterator itr;
301  dbTrackGrid* tg = nullptr;
302 
303  Rect maxRect;
304  if ((extRect.dx() > 0) && (extRect.dy() > 0)) {
305  maxRect = extRect;
306  } else {
307  maxRect = _block->getDieArea();
308  if (!((maxRect.dx() > 0) && (maxRect.dy() > 0))) {
309  logger_->error(
310  RCX, 81, "Die Area for the block has 0 size, or is undefined!");
311  }
312  }
313 
314  if (USE_DB_UNITS) {
315  GetDBcoords2(maxRect);
316  GetDBcoords2(extRect);
317  }
318 
319  std::vector<int> trackXY(32000);
320  uint n = 0;
321  for (itr = layers.begin(); itr != layers.end(); ++itr) {
322  dbTechLayer* layer = *itr;
323 
324  if (layer->getRoutingLevel() == 0) {
325  continue;
326  }
327 
328  n = layer->getRoutingLevel();
329  int w = GetDBcoords2(layer->getWidth());
330  widthTable[n] = layer->getWidth();
331 
332  if (USE_DB_UNITS) {
333  widthTable[n] = w;
334  }
335 
336  W[n] = 1;
337  int s = GetDBcoords2(layer->getSpacing());
338  S[n] = layer->getSpacing();
339 
340  if (USE_DB_UNITS) {
341  S[n] = s;
342  }
343 
344  int p = GetDBcoords2(layer->getPitch());
345  pitchTable[n] = layer->getPitch();
346 
347  if (USE_DB_UNITS) {
348  pitchTable[n] = p;
349  }
350  if (pitchTable[n] <= 0) {
351  logger_->error(RCX,
352  82,
353  "Layer {}, routing level {}, has pitch {}!!",
354  layer->getConstName(),
355  n,
356  pitchTable[n]);
357  }
358 
359  dirTable[n] = 0;
360  if (layer->getDirection() == dbTechLayerDir::HORIZONTAL) {
361  dirTable[n] = 1;
362  }
363 
364  if (skipBaseCalc) {
365  continue;
366  }
367 
368  tg = _block->findTrackGrid(layer);
369  if (tg) {
370  tg->getGridX(trackXY);
371  X1[n] = trackXY[0] - layer->getWidth() / 2;
372  tg->getGridY(trackXY);
373  Y1[n] = trackXY[0] - layer->getWidth() / 2;
374  } else {
375  X1[n] = maxRect.xMin();
376  Y1[n] = maxRect.yMin();
377  }
378  }
379  uint layerCnt = n + 1;
380 
381  _search = new Ath__gridTable(&maxRect, 2, layerCnt, W, pitchTable, S, X1, Y1);
382  _search->setBlock(_block);
383  return layerCnt;
384 }

499  uint wtype,
500  uint step)

506  for (itr = swires.begin(); itr != swires.end(); ++itr) {

519  uint level = s->getTechLayer()->getRoutingLevel();
520 
521  if (step > 0) {

524  len = r.dy();

526

528  _search->addBox(r.xMin(),
529  r.yMin(),
530  r.xMax(),
531  r.yMax(),
532  level,
533  s->getId(),
534  0,

536  } else {
537  if (r.dx() < r.dy()) { // vertical
538  for (int y1 = r.yMin(); y1 < r.yMax();) {
539  int y2 = y1 + step;
540  if (y2 > r.yMax()) {
541  y2 = r.yMax();
542  }
543 
544  _search->addBox(

547  }
548  } else { // horizontal

552  x2 = r.xMax();
553  }
554 
555  _search->addBox(
556  x1, r.yMin(), x2, r.yMax(), level, s->getId(), 0, wtype);
557  x1 = x2;
558  }
559  }
560  }
561  } else {
562  _search->addBox(r.xMin(),
563  r.yMin(),
564  r.xMax(),
565  r.yMax(),
566  level,
567  s->getId(),
568  0,
569  wtype);

572  cnt++;
573  }
574  }
