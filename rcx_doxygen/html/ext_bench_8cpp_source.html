65 uint extRCModel::benchWithVar_density(extMainOptions* opt, extMeasure* measure)
66 {
67  if (opt->_db_only) {
68  return benchDB_WS(opt, measure);
69  }
70  if (opt->_listsFlag) {
71  return benchWithVar_lists(opt, measure);
72  }
73 
74  uint cnt = 0;
75  int met = measure->_met;
76  extVariation* xvar = _process->getVariation(measure->_met);
77 
78  extConductor* cond = _process->getConductor(met);
79  double t = cond->_thickness;
80  double h = cond->_height;
81  double ro = cond->_p;
82 
83  for (uint ii = 0; ii < opt->_widthTable.getCnt(); ii++) {
84  double w = 0.001 * measure->_minWidth * opt->_widthTable.get(ii); // layout
85  measure->_wIndex = measure->_widthTable.findNextBiggestIndex(w); // layout
86 
87  for (uint jj = 0; jj < opt->_spaceTable.getCnt(); jj++) {
88  double s
89  = 0.001 * measure->_minSpace * opt->_spaceTable.get(jj); // layout
90  measure->_sIndex
91  = measure->_widthTable.findNextBiggestIndex(s); // layout
92 
93  for (uint kk = 0; kk < opt->_densityTable.getCnt(); kk++) {
94  double r = opt->_densityTable.get(kk); // layout
95  measure->_rIndex
96  = measure->_dataTable.findNextBiggestIndex(r); // layout
97 
98  double top_width;
99  double top_widthR;
100  double bot_width = w;
101  double thickness = t;
102  double bot_widthR = w;
103  double thicknessR = t;
104 
105  if (r <= 0.0) {
106  top_width = w;
107  top_widthR = w;
108  } else if (xvar != nullptr) {
109  top_width = xvar->getTopWidth(measure->_wIndex, measure->_sIndex);
110  top_widthR = xvar->getTopWidthR(measure->_wIndex, measure->_sIndex);
111 
112  bot_width = xvar->getBottomWidth(w, measure->_rIndex);
113  bot_width = top_width - bot_width;
114 
115  thickness = xvar->getThickness(w, measure->_rIndex);
116 
117  bot_widthR = xvar->getBottomWidthR(w, measure->_rIndex);
118  bot_widthR = top_widthR - bot_widthR;
119 
120  thicknessR = xvar->getThicknessR(w, measure->_rIndex);
121  } else {
122  continue;
123  }
124 
125  measure->setTargetParams(w, s, r, t, h);
126  measureResistance(measure, ro, top_widthR, bot_widthR, thicknessR);
127  measurePatternVar(measure,
128  top_width,
129  bot_width,
130  thickness,
131  measure->_wireCnt,
132  nullptr);
133 
134  cnt++;
135  }
136  }
137  }
138  return cnt;
139 }

141 uint extRCModel::benchWithVar_lists(extMainOptions* opt, extMeasure* measure)
142 {
143  Ath__array1D<double>* wTable = &opt->_widthTable;
144  Ath__array1D<double>* sTable = &opt->_spaceTable;
145  Ath__array1D<double>* thTable = &opt->_thicknessTable;
146  Ath__array1D<double>* gTable = &opt->_gridTable;
147 
148  uint cnt = 0;
149  int met = measure->_met;
150 
151  extConductor* cond = _process->getConductor(met);
152  double t = cond->_thickness;
153  double h = cond->_height;
154  double ro = cond->_p;
155 
156  dbTechLayer* layer = opt->_tech->findRoutingLayer(met);
157 
158  double minWidth = 0.001 * layer->getWidth();
159  ;
160  double pitch = 0.001 * layer->getPitch();
161 
162  if (opt->_default_lef_rules) { // minWidth, minSpacing, minThickness, pitch
163  // multiplied by grid_list
164  wTable->resetCnt();
165  sTable->resetCnt();
166  wTable->add(minWidth);
167 
168  for (uint ii = 0; ii < gTable->getCnt(); ii++) {
169  double s = minWidth + pitch * (gTable->get(ii) - 1);
170  sTable->add(s);
171  }
172  } else if (opt->_nondefault_lef_rules) {
173  wTable->resetCnt();
174  sTable->resetCnt();
175  dbSet<dbTechNonDefaultRule> nd_rules = opt->_tech->getNonDefaultRules();
176  dbSet<dbTechNonDefaultRule>::iterator nditr;
177  dbTechLayerRule* tst_rule;
178 
179  for (nditr = nd_rules.begin(); nditr != nd_rules.end(); ++nditr) {
180  tst_rule = (*nditr)->getLayerRule(layer);
181  if (tst_rule == nullptr) {
182  continue;
183  }
184 
185  double w = tst_rule->getWidth();
186  double s = tst_rule->getSpacing();
187  wTable->add(w);
188  sTable->add(s);
189  }
190  }
191 
192  for (uint ii = 0; ii < wTable->getCnt(); ii++) {
193  double w = wTable->get(ii); // layout
194  for (uint iii = 0; iii < wTable->getCnt(); iii++) {
195  double w2 = wTable->get(iii);
196  for (uint jj = 0; jj < sTable->getCnt(); jj++) {
197  double s = sTable->get(jj); // layout
198  for (uint jjj = 0; jjj < sTable->getCnt(); jjj++) {
199  double s2 = sTable->get(jjj);
200 
201  for (uint kk = 0; kk < thTable->getCnt(); kk++) {
202  double tt = thTable->get(kk); // layout
203  if (!opt->_thListFlag) { // multiplier
204  tt *= t;
205  }
206 
207  double top_width = w;
208  double top_widthR = w;
209 
210  double bot_width = w;
211  double bot_widthR = w;
212 
213  double thickness = tt;
214  double thicknessR = tt;
215 
216  measure->setTargetParams(w, s, 0.0, t, h, w2, s2);
217  measureResistance(measure, ro, top_widthR, bot_widthR, thicknessR);
218  measurePatternVar(measure,
219  top_width,
220  bot_width,
221  thickness,
222  measure->_wireCnt,
223  nullptr);
224 
225  cnt++;
226  }
227  }
228  }
229  }
230  }
231  return cnt;

234 uint extRCModel::linesOverBench(extMainOptions* opt)
235 {
236  if (opt->_met == 0) {
237  return 0;
238  }
239 
240  extMeasure measure(logger_);
241  measure.updateForBench(opt, _extMain);
242  measure._diag = false;
243 
244  sprintf(_patternName, "O%d", opt->_wireCnt + 1);
245  if (opt->_res_patterns) {
246  sprintf(_patternName, "R%d", opt->_wireCnt + 1);
247  }
248 
249  uint cnt = 0;
250 
251  for (int met = 1; met <= (int) _layerCnt; met++) {
252  if (met > opt->_met_cnt) {
253  continue;
254  }
255  if ((opt->_met > 0) && (opt->_met != met)) {
256  continue;
257  }
258 
259  measure._met = met;
260 
261  if (!opt->_db_only) {
262  computeTables(&measure, opt->_wireCnt + 1, 1000, 1000, 1000);
263  }
264 
265  uint patternSep = measure.initWS_box(opt, 20);
266 
267  for (int underMet = 0; underMet < met; underMet++) {
268  if ((opt->_underMet > 0) && (opt->_underMet != underMet)) {
269  continue;
270  }
271 
272  if (met - underMet > (int) opt->_underDist) {
273  continue;
274  }
275 
276  measure.setMets(met, underMet, -1);
277 
278  uint cnt1 = benchWithVar_density(opt, &measure);
279 
280  cnt += cnt1;
281  measure._ur[measure._dir] += patternSep;
282 
283  if (opt->_underMet == 0 && !opt->_gen_def_patterns) {
284  break;
285  }
286 
287  if (underMet == 0 && opt->_res_patterns) {
288  break;
289  }
290  }
291  opt->_ur[0] = std::max(opt->_ur[0], measure._ur[0]);
292  opt->_ur[1] = std::max(opt->_ur[1], measure._ur[1]);
293  }
294 
295  logger_->info(
296  RCX, 55, "Finished {} bench measurements for pattern MET_OVER_MET", cnt);
297 
298  return cnt;
