142 void extProcess::writeGround(FILE* fp,
143  int met,
144  const char* name,
145  double width,
146  double x1,
147  double volt,
148  bool diag)
149 {
150  if (met < 0) {
151  return;
152  }
153  if (!met && diag) {
154  return;
155  }
156 
157  double y1 = -1.0;
158  double th = 1.0;
159  extMasterConductor* m;
160  if (met > 0 && !diag) {
161  m = getMasterConductor(met);
162  y1 = m->_loLeft[2];
163  th = m->_hiLeft[2] - y1;
164  m->writeRaphaelConformalGround(fp, x1, width, this);
165  fprintf(fp, "POLY NAME= M%d__%s; ", met, name);
166  } else {
167  fprintf(fp, "POLY NAME= M0__%s; ", name);
168  }
169 
170  fprintf(fp, " COORD= ");
171 
172  writeRaphaelPointXY(fp, x1, y1);
173  writeRaphaelPointXY(fp, x1 + width, y1);
174  writeRaphaelPointXY(fp, x1 + width, y1 + th);
175  writeRaphaelPointXY(fp, x1, y1 + th);
176 
177  fprintf(fp, " VOLT=%g ;\n", volt);
178 
179  fprintf(fp, "OPTIONS SET_GRID=10000;\n\n");
180 }

249 extMasterConductor::extMasterConductor(uint condId,
250  extConductor* cond,
251  double prevHeight,
252  Logger* logger)
253 {
254  _condId = condId;
255  logger_ = logger;
256  // X coordinates
257 
258  double min_width = cond->_min_width;
259  if (cond->_bottom_right_x - cond->_bottom_left_x > 0) {
260  _loLeft[0] = cond->_bottom_left_x;
261  _loRight[0] = cond->_bottom_right_x;
262  } else {
263  if (cond->_bot_ext == 0.0) {
264  _loLeft[0] = 0;
265  } else {
266  _loLeft[0] = -cond->_bot_ext;
267  }
268  if (!(min_width > 0)) {
269  logger_->warn(RCX,
270  158,
271  "Can't determine Bottom Width for Conductor <{}>",
272  cond->_name);
273  exit(0);
274  }
275  if (cond->_bot_ext == 0.0) {
276  _loRight[0] = min_width;
277  } else {
278  _loRight[0] = min_width + cond->_bot_ext;
279  }
280  }
281  _hiLeft[0] = cond->_top_left_x;
282  _hiRight[0] = cond->_top_right_x;
283 
284  if (cond->_top_right_x - cond->_top_left_x > 0) {
285  _hiLeft[0] = cond->_top_left_x;
286  _hiRight[0] = cond->_top_right_x;
287  } else {
288  if (cond->_top_ext == 0.0) {
289  _hiLeft[0] = 0;
290  } else {
291  _hiLeft[0] = -cond->_top_ext;
292  }
293  if (!(min_width > 0)) {
294  logger_->warn(RCX,
295  152,
296  "Can't determine Top Width for Conductor <{}>",
297  cond->_name);
298  exit(0);
299  }
300  if (cond->_top_ext == 0.0) {
301  _hiRight[0] = min_width;
302  } else {
303  _hiRight[0] = min_width + cond->_top_ext;
304  }
305  }
306 
307  // Y coordinates
308  _loLeft[1] = 0;
309  _loRight[1] = 0;
310  _hiLeft[1] = 0;
311  _hiRight[1] = 0;
312 
313  // Z coordinates
314  double height = cond->_height;
315  if (height <= 0) {
316  height += prevHeight + cond->_distance;
317  cond->_height = height;
318  }
319 
320  _loLeft[2] = height;
321  _loRight[2] = height;
322 
323  double thickness = cond->_thickness;
324  if (!(thickness > 0)) {
325  logger_->warn(
326  RCX, 153, "Can't determine thickness for Conductor <{}>", cond->_name);
327  exit(0);
328  }
329  _hiLeft[2] = height + thickness;
330  _hiRight[2] = height + thickness;
331 
332  _dy = 0;
333  _e = 0.0;
334  for (uint i = 0; i < 3; i++) {
335  _conformalId[i] = 0;
336  }
337 }

410 void extMasterConductor::writeRaphaelConformalGround(FILE* fp,
411  double X,
412  double width,
413  extProcess* p)
414 {
415  double height[3];
416  double thickness[3];
417  double e[3];
418  bool trench = false;
419  uint cnt = 0;
420  uint start = 0;
421  double h = _loLeft[2];
422  for (uint i = 0; i < 3; i++) {
423  uint j = 2 - i;
424  if (!_conformalId[j]) {
425  continue;
426  }
427  if (!start) {
428  start = i;
429  }
430  extDielectric* d = p->getDielectric(_conformalId[j]);
431  // assuming conformal and trench will not show up at the same time. Also
432  // height for the trench layer is negative.
433  trench = d->_trench;
434  height[i] = d->_height;
435  thickness[i] = d->_thickness;
436  e[i] = d->_epsilon;
437  h += d->_thickness;
438  cnt++;
439  }
440  if (!cnt) {
441  return;
442  }
443  if (trench) {
444  for (uint j = start; j < start + cnt; j++) {
445  fprintf(fp, "POLY NAME=");
446  fprintf(fp, "M%d_Trench%d;", _condId, 2 - j);
447 
448  writeRaphaelPointXY(fp, X, _hiRight[2] + height[j]);
449  writeRaphaelPointXY(fp, X + width, _hiRight[2] + height[j]);
450  writeRaphaelPointXY(fp, X + width, _hiRight[2]);
451  writeRaphaelPointXY(fp, X, _hiRight[2]);
452  fprintf(fp, " DIEL=%g\n", e[j]);
453  }
454  return;
455  }
456  for (uint j = start; j < start + cnt; j++) {
457  fprintf(fp, "POLY NAME=");
458  fprintf(fp, "M%d_Conformal%d;", _condId, 2 - j);
459 
460  h -= thickness[j];
461  writeRaphaelPointXY(fp, X, h);
462  writeRaphaelPointXY(fp, X + width, h);
463  writeRaphaelPointXY(fp, X + width, h + height[j]);
464  writeRaphaelPointXY(fp, X, h + height[j]);
465  fprintf(fp, " DIEL=%g\n", e[j]);
466  }
467 }

556 double extProcess::adjustMasterLayersForHeight(uint met, double thickness)
557 {
558  double condThickness = _condTable->get(met)->_thickness;
559  double dth = (thickness - condThickness) / condThickness;
560 
561  double h = 0.0;
562  for (uint ii = 1; ii < _masterConductorTable->getCnt(); ii++) {
563  extConductor* cond = _condTable->get(ii);
564  extMasterConductor* m = _masterConductorTable->get(ii);
565 
566  if (_thickVarFlag) {
567  h += cond->_distance * (1 + dth);
568  } else {
569  h += cond->_distance;
570  }
571 
572  double th = cond->_thickness;
573  if (_thickVarFlag) {
574  th *= (1 + dth);
575  } else if (ii == met) {
576  th = thickness;
577  }
578 
579  m->resetThicknessHeight(h, th);
580  h += th;
581  }
582  return dth;
583 }
