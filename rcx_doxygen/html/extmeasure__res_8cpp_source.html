39 int extMeasure::computeResDist(SEQ* s,
40  uint trackMin,
41  uint trackMax,
42  uint targetMet,
43  Ath__array1D<SEQ*>* diagTable)
44 {
45  int trackDist = 4;
46  int loTrack;
47  int hiTrack;
48  int planeIndex
49  = getDgPlaneAndTrackIndex(targetMet, trackDist, loTrack, hiTrack);
50  if (planeIndex < 0) {
51  return 0;
52  }
53 
54  uint len = 0;
55 
56  Ath__array1D<SEQ*> tmpTable(16);
57  copySeqUsingPool(s, &tmpTable);
58 
59  Ath__array1D<SEQ*> residueTable(16);
60 
61  int trackTable[200];
62  uint cnt = 0;
63  for (int kk = (int) trackMin; kk <= (int) trackMax;
64  kk++) // skip overlapping track
65  {
66  if (-kk >= _dgContextLowTrack[planeIndex]) {
67  trackTable[cnt++] = *_dgContextTracks / 2 - kk;
68  }
69  }
70 
71  for (uint ii = 0; ii < cnt; ii++) {
72  int trackn = trackTable[ii];
73 
74  if (_dgContextArray[planeIndex][trackn]->getCnt() <= 1) {
75  continue;
76  }
77  // Check for same track
78  bool same_track = false;
79  Ath__array1D<SEQ*>* dTable = _dgContextArray[planeIndex][trackn];
80  int cnt = (int) dTable->getCnt();
81  for (uint idx = 1; idx < (int) cnt; idx++) {
82  SEQ* tseq = dTable->get(idx);
83  if (tseq->type == _rsegSrcId) {
84  same_track = true;
85  break;
86  }
87  }
88  if (same_track) {
89  continue;
90  }
91 
92  bool add_all_diag = false;
93  if (!add_all_diag) {
94  for (uint jj = 0; jj < tmpTable.getCnt(); jj++) {
95  len += computeRes(tmpTable.get(jj),
96  targetMet,
97  _dir,
98  planeIndex,
99  trackn,
100  &residueTable);
101  }
102  } else {
103  len += computeRes(s, targetMet, _dir, planeIndex, trackn, &residueTable);
104  }
105 
106  seq_release(&tmpTable);
107  tableCopyP(&residueTable, &tmpTable);
108  residueTable.resetCnt();
109  }
110  if (diagTable != nullptr) {
111  tableCopyP(&tmpTable, diagTable);
112  } else {
113  seq_release(&tmpTable);
114  }
115  return len;
116 }

118 uint extMeasure::computeRes(SEQ* s,
119  uint targetMet,
120  uint dir,
121  uint planeIndex,
122  uint trackn,
123  Ath__array1D<SEQ*>* residueSeq)
124 {
125  Ath__array1D<SEQ*>* dgContext = _dgContextArray[planeIndex][trackn];
126  if (dgContext->getCnt() <= 1) {
127  return 0;
128  }
129 
130  Ath__array1D<SEQ*> overlapSeq(16);
131  getDgOverlap_res(s, _dir, dgContext, &overlapSeq, residueSeq);
132 
133  uint len = 0;
134  for (uint jj = 0; jj < overlapSeq.getCnt(); jj++) {
135  SEQ* tgt = overlapSeq.get(jj);
136  uint diagDist = calcDist(tgt->_ll, tgt->_ur);
137  uint len1 = getLength(tgt, !_dir);
138 
139  DebugDiagCoords(_met, targetMet, len1, diagDist, tgt->_ll, tgt->_ur);
140  len += len1;
141  calcRes(_rsegSrcId, len1, _dist, diagDist, _met);
142  }
143  seq_release(&overlapSeq);
144  return len;
145 }

171 void extMeasure::calcRes(int rsegId1,
172  uint len,
173  int dist1,
174  int dist2,
175  int tgtMet)
176 {
177  if (dist1 == -1) {
178  dist1 = 0;
179  }
180  if (dist2 == -1) {
181  dist1 = 0;
182  }
183  int min_dist = 0;
184  if (dist1 > dist2) {
185  min_dist = dist1;
186  dist1 = dist2;
187  dist2 = min_dist;
188  }
189  uint modelCnt = _metRCTable.getCnt();
190  for (uint ii = 0; ii < modelCnt; ii++) {
191  extMetRCTable* rcModel = _metRCTable.get(ii);
192  if (rcModel->_resOver[tgtMet] == nullptr) {
193  continue;
194  }
195 
196  extDistRC* rc = rcModel->_resOver[tgtMet]->getRes(0, _width, dist1, dist2);
197  if (rc != nullptr) {
198  double R1 = rc->_diag > 0 ? rc->_diag : rc->_res;
199  double R = len * R1;
200  _rc[ii]->_res += R;
201  }
202  }

248 extDistRC* extDistRCTable::getComputeRC_res(uint dist1, uint dist2)
249 {
250  int min_dist = 0;
251  if (dist1 > dist2) {
252  min_dist = dist1;
253  dist1 = dist2;
254  dist2 = min_dist;
255  }
256  if (_measureTable == nullptr) {
257  return nullptr;
258  }
259 
260  if (_measureTable->getCnt() <= 0) {
261  return nullptr;
262  }
263 
264  extDistRC* rc1 = _measureTableR[0]->geti(0);
265  rc1->_diag = 0.0;
266  if (rc1 == nullptr) {
267  return nullptr;
268  }
269 
270  if (dist1 + dist2 == 0) { // ASSUMPTION: 0 dist exists as first
271  return rc1;
272  }
273  if (dist1 >= _maxDist && dist2 >= _maxDist) {
274  return nullptr;
275  }
276  if (dist2 > _maxDist) {
277  dist2 = dist1;
278  dist1 = 0;
279  }
280 
281  uint index_dist = 0;
282  bool found = false;
283  extDistRC* rc2 = _measureTableR[1]->geti(0);
284  if (rc2 == nullptr) {
285  return rc1;
286  }
287 
288  if (dist1 <= rc1->_sep) {
289  index_dist = 0;
290  found = true;
291  } else if (dist1 < rc2->_sep) {
292  index_dist = 0;
293  found = true;
294  } else if (dist1 == rc2->_sep) {
295  index_dist = 1;
296  found = true;
297  } else {
298  index_dist = 2;
299  }
300  extDistRC* rc = nullptr;
301  if (!found) {
302  // find first dist
303 
304  uint ii;
305  for (ii = index_dist; ii < _distCnt; ii++) {
306  rc = _measureTableR[ii]->geti(0);
307  if (dist1 == rc->_sep) {
308  found = true;
309  index_dist = ii;
310  break;
311  }
312  if (dist1 < rc->_sep) {
313  found = true;
314  index_dist = ii;
315  break;
316  }
317  }
318  if (!found && ii == _distCnt) {
319  index_dist = ii - 1;
320  found = true;
321  }
322  }
323  if (found) {
324  _measureTable = _measureTableR[index_dist];
325  _computeTable = _computeTableR[index_dist];
326  extDistRC* res = findIndexed_res(dist1, dist2);
327  res->_diag = 0;
328  if (rc != nullptr && dist1 < rc->_sep) {
329  _measureTable = _measureTableR[index_dist - 1];
330  _computeTable = _computeTableR[index_dist - 1];
331  extDistRC* res1 = findIndexed_res(dist1, dist2);
332  double R1 = res->interpolate_res(dist1, res1);
333  res1->_diag = R1;
334  return res1;
335  }
336  return res;
337  }
338  return nullptr;
339 }

383 void extMeasure::getDgOverlap_res(SEQ* sseq,
384  uint dir,
385  Ath__array1D<SEQ*>* dgContext,
386  Ath__array1D<SEQ*>* overlapSeq,
387  Ath__array1D<SEQ*>* residueSeq)
388 {
389  int idx = 1;
390  uint lp = dir ? 0 : 1;
391  uint wp = dir ? 1 : 0;
392  SEQ* rseq;
393  SEQ* tseq;
394  SEQ* wseq;
395  int covered = sseq->_ll[lp];
396 
397  if (idx == dgContext->getCnt()) {
398  rseq = _seqPool->alloc();
399  rseq->_ll[wp] = sseq->_ll[wp];
400  rseq->_ur[wp] = sseq->_ur[wp];
401  rseq->_ll[lp] = sseq->_ll[lp];
402  rseq->_ur[lp] = sseq->_ur[lp];
403  residueSeq->add(rseq);
404  return;
405  }
406 
407  for (; idx < (int) dgContext->getCnt(); idx++) {
408  tseq = dgContext->get(idx);
409  if (tseq->_ur[lp] <= covered) {
410  continue;
411  }
412 
413  if (tseq->_ll[lp] >= sseq->_ur[lp]) {
414  rseq = _seqPool->alloc();
415  rseq->_ll[wp] = sseq->_ll[wp];
416  rseq->_ur[wp] = sseq->_ur[wp];
417  rseq->_ll[lp] = covered;
418  rseq->_ur[lp] = sseq->_ur[lp];
419  assert(rseq->_ur[lp] >= rseq->_ll[lp]);
420  residueSeq->add(rseq);
421  break;
422  }
423  wseq = _seqPool->alloc();
424  wseq->type = tseq->type;
425  wseq->_ll[wp] = tseq->_ll[wp];
426  wseq->_ur[wp] = tseq->_ur[wp];
427  if (tseq->_ur[lp] <= sseq->_ur[lp]) {
428  wseq->_ur[lp] = tseq->_ur[lp];
429  } else {
430  wseq->_ur[lp] = sseq->_ur[lp];
431  }
432  if (tseq->_ll[lp] <= covered) {
433  wseq->_ll[lp] = covered;
434  } else {
435  wseq->_ll[lp] = tseq->_ll[lp];
436  rseq = _seqPool->alloc();
437  rseq->_ll[wp] = sseq->_ll[wp];
438  rseq->_ur[wp] = sseq->_ur[wp];
439  rseq->_ll[lp] = covered;
440  rseq->_ur[lp] = tseq->_ll[lp];
441  assert(rseq->_ur[lp] >= rseq->_ll[lp]);
442  residueSeq->add(rseq);
443  }
444  assert(wseq->_ur[lp] >= wseq->_ll[lp]);
445  overlapSeq->add(wseq);
446  covered = wseq->_ur[lp];
447  if (tseq->_ur[lp] >= sseq->_ur[lp]) {
448  break;
449  }
450  if (idx == (int) dgContext->getCnt() - 1 && covered < sseq->_ur[lp]) {
451  rseq = _seqPool->alloc();
452  rseq->_ll[wp] = sseq->_ll[wp];
453  rseq->_ur[wp] = sseq->_ur[wp];
454  rseq->_ll[lp] = covered;
455  rseq->_ur[lp] = sseq->_ur[lp];
456  assert(rseq->_ur[lp] >= rseq->_ll[lp]);
457  residueSeq->add(rseq);
458  }
459  }
460  dgContext->get(0)->_ll[0] = idx;
461  if (idx == dgContext->getCnt() && residueSeq->getCnt() == 0) {
462  rseq = _seqPool->alloc();
463  rseq->_ll[wp] = sseq->_ll[wp];
464  rseq->_ur[wp] = sseq->_ur[wp];
465  rseq->_ll[lp] = sseq->_ll[lp];
466  rseq->_ur[lp] = sseq->_ur[lp];
467  residueSeq->add(rseq);
468  }
469 }
