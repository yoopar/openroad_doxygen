85 
86  set skip false
87  set line 1

92  if {$ignore ne ""} {
93  set skip [regexp $ignore $line1 || regexp $ignore $line2]
94  }
95  if { !$skip && $line1 != $line2 } {
96  set found_diff 1

98  }
99  incr line
100  set line1_length [gets $stream1 line1]
101  set line2_length [gets $stream2 line2]
102  }
103  close $stream1
104  close $stream2
105  if { $found_diff || $line1_length != $line2_length} {
106  puts "Differences found at line $line."
107  puts "$line1"
108  puts "$line2"

110  } else {
