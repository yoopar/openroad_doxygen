69 def extract_arguments(text):
70  # Goal is to extract all the text from the end of tcl code to the next ### header.
71  # Returns options and arguments.
72  level2 = extract_headers(text, 2)
73  level3 = extract_headers(text, 3)
74 
75  # form these 2 regex styles.
76  # ### Header 1 {text} ### Header2; ### Header n-2 {text} ### Header n-1
77  # ### Header n {text} ## closest_level2_header
78  first = [rf'### ({level3[i]})(.*?)### ({level3[i+1]})' for i in range(len(level3) - 1)]
79 
80  # find the next closest level2 header to the last level3 header.
81  closest_level2 = [text.find(f"## {x}") - text.find(f"### {level3[-1]}") for x in level2]
82  closest_level2_idx = [idx for idx, x in enumerate(closest_level2) if x > 0][0]
83 
84  # This will disambiguate cases where different level headers share the same name.
85  second = [rf"### ({level3[-1]})(.*?)## ({level2[closest_level2_idx]})"]
86  final_options, final_args = [], []
87  for idx, regex in enumerate(first + second):
88  match = re.findall(regex, text, flags = re.DOTALL)
89  #print(regex)
90  # get text until the next header
91  a = match[0][1]
92  a = a[a.find("#"):]
93  options = a.split("####")[1:]
94  if not options:
95  final_options.append([])
96  final_args.append([])
97  continue
98  options, args = options[0], options[1:]
99  final_options.append(extract_tables(options))
100  tmp_arg = []
101  for arg in args:
102  tmp_arg.extend(extract_tables(arg))
103  final_args.append(tmp_arg)
104  return final_options, final_args
