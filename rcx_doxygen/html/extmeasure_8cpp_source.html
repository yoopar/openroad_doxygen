337 uint extMeasure::createContextNets(char* dirName,
338  const int bboxLL[2],
339  const int bboxUR[2],
340  int met,
341  double pitchMult)
342 {
343  if (met <= 0) {
344  return 0;
345  }
346 
347  dbTechLayer* layer = _tech->findRoutingLayer(met);
348  dbTechLayer* mlayer = _tech->findRoutingLayer(_met);
349  uint minWidth = layer->getWidth();
350  uint minSpace = layer->getSpacing();
351  int pitch = lround(1000 * ((minWidth + minSpace) * pitchMult) / 1000);
352 
353  int ll[2];
354  int ur[2];
355 
356  uint offset = 0;
357  ll[_dir] = bboxLL[_dir] - offset;
358  ur[_dir] = bboxUR[_dir] + offset;
359 
360  _ur[_dir] = ur[_dir];
361 
362  uint cnt = 1;
363 
364  uint not_dir = !_dir;
365  int start = bboxLL[not_dir] + offset;
366  int end = bboxUR[not_dir];
367  for (int lenXY = (int) (start + minWidth); (int) (lenXY + minWidth) <= end;
368  lenXY += pitch) {
369  ll[not_dir] = lenXY;
370  ur[not_dir] = lenXY + minWidth;
371 
372  char netName[1024];
373  sprintf(netName, "%s_m%d_cntxt_%d", dirName, met, cnt++);
374  dbNet* net;
375  assert(_create_net_util.getBlock() == _block);
376  if (mlayer->getDirection() != dbTechLayerDir::HORIZONTAL) {
377  net = _create_net_util.createNetSingleWire(netName,
378  ll[0],
379  ll[1],
380  ur[0],
381  ur[1],
382  met,
383  dbTechLayerDir::HORIZONTAL,
384  false);
385  } else {
386  net = _create_net_util.createNetSingleWire(netName,
387  ll[0],
388  ll[1],
389  ur[0],
390  ur[1],
391  met,
392  mlayer->getDirection(),
393  false);
394  }
395  addNew2dBox(net, ll, ur, met, true);
396  }
397  return cnt - 1;
398 }
