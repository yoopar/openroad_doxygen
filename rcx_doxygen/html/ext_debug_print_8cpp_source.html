128 bool extMeasure::printTraceNet(const char* msg,
129  bool init,
130  dbCCSeg* cc,
131  uint overSub,
132  uint covered)
133 {
134  if (!IsDebugNet()) {
135  return false;
136  }
137 
138  if (init) {
139  if (overSub + covered > 0) {
140  fprintf(_debugFP, "SUB %d GS %d ", overSub, covered);
141  }
142 
143  if (_netSrcId == _netId) {
144  printTraceNetInfo("", _netSrcId, _rsegSrcId);
145  } else {
146  printTraceNetInfo("", _netTgtId, _rsegTgtId);
147  }
148 
149  return true;
150  }
151 
152  fprintf(_debugFP, "%s ", msg);
153  if (overSub + covered > 0) {
154  fprintf(_debugFP, " L %d SUB %d GS %d ", _len, overSub, covered);
155  }
156 
157  if (cc != nullptr) {
158  fprintf(_debugFP, " %g ", cc->getCapacitance());
159  }
160 
161  printTraceNetInfo("", _netSrcId, _rsegSrcId);
162  printTraceNetInfo(" ", _netTgtId, _rsegTgtId);
163  fprintf(_debugFP, "\n");
164  return true;
