63 gs::gs(odb::AthPool<SEQ>* pool)
64 {
65  init_ = INIT;
66 
67  pixint sum = PIXFILL;
68  for (int i = 0; i < PIXMAPGRID; i++) {
69  start_[i] = sum;
70  sum = (sum >> 1);
71  }
72 
73  sum = PIXMAX;
74  pixint s2 = sum;
75  for (int i = 0; i < PIXMAPGRID; i++) {
76  end_[i] = sum;
77  sum = (sum >> 1) | PIXMAX;
78 
79  middle_[i] = s2;
80  s2 = (s2 >> 1);
81  }
82 
83  nslices_ = -1;
84  maxslice_ = -1;
85 
86  seqPool_ = pool;
87 }

136 void gs::setSize(int pl, int xres, int yres, int x0, int y0, int x1, int y1)
137 {
138  plc_ = pldata_[pl];
139 
140  plc_->x0 = x0;
141  plc_->x1 = x1;
142  plc_->y0 = y0;
143  plc_->y1 = y1;
144 
145  if (plc_->x1 <= plc_->x0) {
146  // to avoid things like divide by 0, etc
147  plc_->x1 = plc_->x0 + 1;
148  }
149 
150  if (plc_->y1 <= plc_->y0) {
151  // to avoid things like divide by 0, etc
152  plc_->y1 = plc_->y0 + 1;
153  }
154 
155  plc_->xres = xres;
156  plc_->yres = yres;
157 
158  plc_->width = (plc_->x1 - plc_->x0 + 1) / plc_->xres;
159  if (((plc_->x1 - plc_->x0 + 1) % plc_->xres) != 0) {
160  plc_->width++;
161  }
162 
163  plc_->height = (plc_->y1 - plc_->y0 + 1) / plc_->yres;
164  if (((plc_->y1 - plc_->y0 + 1) % plc_->yres) != 0) {
165  plc_->height++;
166  }
167 
168  plc_->pixwrem = plc_->width % PIXMAPGRID;
169 
170  // how many pixels pixmap is wide, upped to multiple of PIXMAPGRID
171  int pixwidth = plc_->width;
172  if (plc_->pixwrem != 0) {
173  pixwidth += (PIXMAPGRID - plc_->pixwrem);
174  }
175 
176  plc_->pixstride = pixwidth / PIXMAPGRID;
177 
178  plc_->pixfullblox = plc_->pixstride;
179  if (plc_->pixwrem != 0) {
180  plc_->pixfullblox--;
181  }
182 
183  pixmap* pm = (pixmap*) calloc(plc_->height * plc_->pixstride + PIXADJUST,
184  sizeof(pixmap));
185  if (pm == nullptr) {
186  fprintf(stderr,
187  "Error: not enough memory available trying to allocate plane %d\n",
188  pl);
189  exit(-1);
190  }
191 
192  plc_->plalloc = pm;
193  plc_->plane = pm;
194 }

214 int gs::box(int px0, int py0, int px1, int py1, int sl)
215 {
216  if ((sl < 0) || (sl > nslices_)) {
217  fprintf(stderr,
218  "Box in slice %d exceeds maximum configured slice count %d - "
219  "ignored!\n",
220  sl,
221  nslices_);
222  return -1;
223  }
224 
225  if (!(init_ & GS_ALL)) {
226  return -1;
227  }
228 
229  if (sl > maxslice_) {
230  maxslice_ = sl;
231  }
232 
233  plc_ = pldata_[sl];
234 
235  // normalize bbox
236  if (px0 > px1) {
237  std::swap(px0, px1);
238  }
239  if (py0 > py1) {
240  std::swap(py0, py1);
241  }
242 
243  if (px1 < plc_->x0) {
244  return -1;
245  }
246  if (px0 > plc_->x1) {
247  return -1;
248  }
249  if (py1 < plc_->y0) {
250  return -1;
251  }
252  if (py0 > plc_->y1) {
253  return -1;
254  }
255 
256  // convert to pixel space
257  int cx0 = int((px0 - plc_->x0) / plc_->xres);
258  int cx1 = int((px1 - plc_->x0) / plc_->xres);
259  int cy0 = int((py0 - plc_->y0) / plc_->yres);
260  int cy1 = int((py1 - plc_->y0) / plc_->yres);
261 
262  // render a rectangle on the selected slice. Paint all pixels
263  cx0 = clip(cx0, 0, plc_->width);
264  cx1 = clip(cx1, 0, plc_->width);
265  cy0 = clip(cy0, 0, plc_->height);
266  cy1 = clip(cy1, 0, plc_->height);
267  // now fill in planes object
268 
269  // xbs = x block start - block the box starts in
270  const int xbs = cx0 / PIXMAPGRID;
271  // xbe = x block end - block the box ends in
272  const int xbe = cx1 / PIXMAPGRID;
273 
274  pixint smask = start_[cx0 % PIXMAPGRID];
275  const pixint emask = end_[cx1 % PIXMAPGRID];
276 
277  if (xbe == xbs) {
278  smask &= emask;
279  }
280 
281  pixmap* pm = plc_->plane + plc_->pixstride * cy0 + xbs;
282 
283  for (int yb = cy0; yb <= cy1; yb++) {
284  // start block
285  pixmap* pcb = pm;
286 
287  // for next time through loop - allow compiler time for out-of-order
288  pm += plc_->pixstride;
289 
290  // do "start" block
291  pcb->lword = pcb->lword | smask;
292 
293  // do "middle" block
294  for (int mb = xbs + 1; mb < xbe;) {
295  pcb++;
296  // moved here to allow for out-of-order execution
297  mb++;
298 
299  pcb->lword = PIXFILL;
300  }
301 
302  // do "end" block
303  if (xbe > xbs) {
304  pcb++;
305  pcb->lword = pcb->lword | emask;
306  }
307  }
308 
309  return 0;
310 }

385 /* get_seq - returns an integer corresponding to the longest uninterrupted
386  * sequence of virtual bits found of the same type (set or unset)
387  *
388  * Parameters: ll - lower left array [0] = x0, [1] = y0
389  * ur - upper right array [0] = x1, [1] = y1
390  * order - search by column or by row (GS_COLUMN, GS_ROW)
391  * plane - which plane to search
392  * array - pool of sequence pointers to get a handle from
393  */
394 
395 uint gs::get_seq(int* ll,
396  int* ur,
397  uint order,
398  uint plane,
399  odb::Ath__array1D<SEQ*>* array)
400 {
401  if (check_slice(plane) != 0) {
402  return 0;
403  }
404 
405  plc_ = pldata_[plane];
406 
407  // Sanity checks
408  if (ur[0] < plc_->x0) {
409  return 0;
410  }
411  if (ll[0] > plc_->x1) {
412  return 0;
413  }
414 
415  if (ur[1] < plc_->y0) {
416  return 0;
417  }
418  if (ll[1] > plc_->y1) {
419  return 0;
420  }
421 
422  if (ll[0] < plc_->x0) {
423  ll[0] = plc_->x0;
424  }
425 
426  if (ur[0] > plc_->x1) {
427  ur[0] = plc_->x1;
428  }
429 
430  if (ll[1] < plc_->y0) {
431  ll[1] = plc_->y0;
432  }
433 
434  if (ur[1] > plc_->y1) {
435  ur[1] = plc_->y1;
436  }
437  // End Sanity Checks
438 
439  SEQ* s = salloc();
440 
441  // convert into internal coordinates
442  const int cx0 = int((ll[0] - plc_->x0) / plc_->xres);
443  const int cy0 = int((ll[1] - plc_->y0) / plc_->yres);
444 
445  int cx1 = int((ur[0] - plc_->x0) / plc_->xres);
446  if (((ur[0] - plc_->x0 + 1) % plc_->xres) != 0) {
447  cx1++;
448  }
449  int cy1 = int((ur[1] - plc_->y0) / plc_->yres);
450  if (((ur[1] - plc_->y0 + 1) % plc_->yres) != 0) {
451  cy1++;
452  }
453 
454  int blacksum = 0;
455 
456  if (order == GS_ROW) {
457  int rs = ll[1];
458  int re = ur[1];
459 
460  for (int row = cy0; row <= cy1; row++) {
461  int start = cx0;
462  int end = cx1;
463  bool flag = false;
464  while (get_seqrow(row, plane, start, end, s->type) == 0) {
465  s->_ll[0] = (int) (start * (plc_->xres) + plc_->x0);
466  s->_ur[0] = (int) ((end + 1) * (plc_->xres) + (plc_->x0) - 1);
467  if (s->_ur[0] >= ur[0]) {
468  s->_ur[0] = ur[0];
469  flag = true;
470  }
471 
472  if ((row == cy0) && (s->_ll[0] < ll[0])) {
473  s->_ll[0] = ll[0];
474  }
475 
476  s->_ll[1] = rs;
477  s->_ur[1] = re;
478 
479  if (s->type == GS_BLACK) {
480  blacksum += (s->_ur[0] - s->_ll[0]);
481  }
482 
483  if (array != nullptr) {
484  array->add(s);
485  s = salloc();
486  }
487  if (flag) {
488  seqPool_->free(s);
489  return blacksum;
490  }
491  start = end + 1;
492  }
493  rs += plc_->yres;
494  if (rs > ur[1]) {
495  break;
496  }
497  re += plc_->yres;
498  }
499  } else if (order == GS_COLUMN) {
500  int cs = ((cx1 + cx0) / 2) * plc_->xres + plc_->x0;
501  int ce = cs;
502 
503  if (cs < ll[0]) {
504  cs = ll[0];
505  }
506 
507  if (ce > ur[0]) {
508  ce = ur[0];
509  }
510 
511  for (int col = cx0; col <= cx0; col++) {
512  int start = cy0;
513  int end;
514  bool flag = false;
515  while (get_seqcol(col, plane, start, end, s->type) == 0) {
516  s->_ll[1] = (int) (start * plc_->yres + plc_->y0);
517  s->_ur[1] = (int) ((end + 1) * plc_->yres + plc_->y0 - 1);
518  if (s->_ur[1] >= ur[1]) {
519  flag = true;
520  s->_ur[1] = ur[1];
521  }
522 
523  if ((col == cx0) && (s->_ll[1] < ll[1])) {
524  s->_ll[1] = ll[1];
525  }
526 
527  s->_ll[0] = cs;
528  s->_ur[0] = ce;
529 
530  if (s->type == GS_BLACK) {
531  blacksum += (s->_ur[1] - s->_ll[1]);
532  }
533 
534  if (array != nullptr) {
535  array->add(s);
536  s = salloc();
537  }
538  if (flag) {
539  seqPool_->free(s);
540  return blacksum;
541  }
542  start = end + 1;
543  }
544  cs += plc_->xres;
545  if (cs > ur[0]) {
546  break;
547  }
548  ce += plc_->xres;
549  }
550  }
551  seqPool_->free(s);
552  return blacksum;
553 }

555 int gs::get_seqrow(const int y,
556  const int plane,
557  const int stpix,
558  int& epix,
559  int& seqcol)
560 {
561  if (!(init_ & ALLOCATED)) {
562  return -1;
563  }
564 
565  if (y >= plc_->height) {
566  return -1;
567  }
568 
569  if (stpix >= plc_->width) {
570  return -1;
571  }
572 
573  int sto = stpix / PIXMAPGRID;
574  const int str = stpix - (sto * PIXMAPGRID);
575 
576  const long offset = y * plc_->pixstride + sto;
577  pixmap* pl = pldata_[plane]->plane + offset;
578 
579  seqcol = GS_NONE;
580 
581  // take care of start
582  if ((pl->lword & start_[str]) == start_[str]) {
583  seqcol = GS_BLACK;
584  sto++;
585  pl++;
586  } else if ((pl->lword & start_[str]) == 0) {
587  seqcol = GS_WHITE;
588  sto++;
589  pl++;
590  } else {
591  // ends here already
592  const pixint bc = (pl->lword) & middle_[str];
593  if (bc != 0) {
594  seqcol = GS_BLACK;
595  } else {
596  seqcol = GS_WHITE;
597  }
598  epix = (sto * PIXMAPGRID) + str;
599  for (int bit = str + 1; bit < PIXMAPGRID; bit++) {
600  if ((pl->lword & middle_[bit]) == 0) {
601  if (seqcol == GS_BLACK) {
602  break;
603  }
604  } else {
605  if (seqcol == GS_WHITE) {
606  break;
607  }
608  }
609 
610  epix++;
611  }
612  return 0;
613  }
614 
615  if (sto > plc_->pixfullblox) {
616  epix = plc_->width - 1;
617  return 0;
618  }
619 
620  int st;
621  for (st = sto; st < plc_->pixfullblox; st++) {
622  if ((pl->lword) == PIXFILL) {
623  if (seqcol != GS_BLACK) {
624  epix = st * PIXMAPGRID - 1;
625  return 0;
626  }
627  } else if (pl->lword == 0) {
628  if (seqcol != GS_WHITE) {
629  epix = st * PIXMAPGRID - 1;
630  return 0;
631  }
632  } else {
633  // ends here
634  epix = st * PIXMAPGRID - 1;
635  for (int bit = 0; bit < PIXMAPGRID; bit++) {
636  if (((pl->lword) & middle_[bit]) == 0) {
637  if (seqcol == GS_BLACK) {
638  break;
639  }
640  } else {
641  if (seqcol == GS_WHITE) {
642  break;
643  }
644  }
645 
646  epix++;
647  }
648  return 0;
649  }
650  pl++;
651  }
652  // handle non-even case
653  epix = st * PIXMAPGRID - 1;
654  for (int bit = 0; bit < plc_->pixwrem; bit++) {
655  if (((pl->lword) & middle_[bit]) == 0) {
656  if (seqcol == GS_BLACK) {
657  break;
658  }
659  } else {
660  if (seqcol == GS_WHITE) {
661  break;
662  }
663  }
664  epix++;
665  }
666  return 0;
667 }

669 int gs::get_seqcol(const int x,
670  const int plane,
671  const int stpix,
672  int& epix,
673  int& seqcol)
674 {
675  if (!(init_ & ALLOCATED)) {
676  return -1;
677  }
678 
679  if (x >= plc_->width) {
680  return -1;
681  }
682 
683  if (stpix >= plc_->height) {
684  return -1;
685  }
686 
687  // get proper "word" offset
688  const int sto = x / PIXMAPGRID;
689  const int stc = x - (sto * PIXMAPGRID);
690 
691  const long offset = sto + stpix * plc_->pixstride;
692  const pixint bitmask = middle_[stc];
693  pixmap* pl = pldata_[plane]->plane + offset;
694 
695  seqcol = GS_NONE;
696 
697  if (pl->lword & bitmask) {
698  seqcol = GS_BLACK;
699  } else {
700  seqcol = GS_WHITE;
701  }
702 
703  for (int row = stpix + 1; row < plc_->height; row++) {
704  pl += plc_->pixstride;
705  if (pl->lword & bitmask) {
706  if (seqcol == GS_BLACK) {
707  continue;
708  }
709  epix = row - 1;
710  return 0;
711 
712  } else {
713  if (seqcol == GS_BLACK) {
714  epix = row - 1;
715  return 0;
716  }
717  continue;
718  }
719  }
720 
721  epix = plc_->height;
722  return 0;
723 }
