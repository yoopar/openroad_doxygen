111 uint extMain::calcMinMaxRC()
112 {
113  _currentModel = getRCmodel(0);
114 
115  uint cnt = 0;
116 
117  for (odb::dbTechLayer* layer : _tech->getLayers()) {
118  if (layer->getRoutingLevel() == 0) {
119  continue;
120  }
121 
122  const int met = layer->getRoutingLevel();
123  const int width = layer->getWidth();
124  int dist = layer->getSpacing();
125  if (dist == 0) {
126  dist = layer->getPitch() - layer->getWidth();
127  }
128 
129  for (uint jj = 0; jj < _modelMap.getCnt(); jj++) {
130  resetMinMaxRC(met, jj);
131 
132  extDistRC* rcMin = _currentModel->getMinRC(met, width);
133  extDistRC* rcMax = _currentModel->getMaxRC(met, width, dist);
134 
135  setMinRC(met, jj, rcMin);
136  setMaxRC(met, jj, rcMax);
137  }
138  cnt++;
139  }
140  return cnt;
141 }

199  len = dx;

202  }

205  _tmpLenStats += buf;
206  wlen += len;
207 
208  min_res += len * _minResTable[met][corner];
209  max_res += len * _maxResTable[met][corner];
210 
211  min_cap += len * _minCapTable[met][corner];
212  max_cap += len * _maxCapTable[met][corner];
213  }
